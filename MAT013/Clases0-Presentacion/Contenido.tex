\subsection{Nota Final}
% \begin{frame}
% \subsectionpage
% \end{frame}

% \begin{frame}
% \tableofcontents[currentsubsection]
% \end{frame}
}

\begin{frame}
  \begin{block}{Nota Semestral}
    $$NS=0,25\cdot NQ+0,20\cdot C_1+0,25\cdot C_2+0,3\cdot C_3$$
  \end{block}
  \vspace{0.25cm}
  \onslide<2->{
  \begin{block}{Nota Final}
    \par Se tendrán dos casos
    \begin{itemize}
      \item si $NS\geq 54,5$ entonces aprueba con nota final 
        $$NF=NS$$
      \item si $NS   < 54,5$ podrá rendir el certamen global.
        % $$NF=0.7\cdot NS+0.3\cdot CG$$
      % \item si $NS<45$ entonces reprueba con nota final
      %   $$NF=NS$$
    \end{itemize}
  \end{block}
  }
  \vspace{0.25cm}
  \onslide<3->{
    En caso de rendir el certamen global el cálculo de la nota final será \textbf{recalculando} la nota $NS$, donde $CG$ sustituye sólo uno de los 3 certámenes rendidos, quedando como Nota Final $NF$ aquel caso en que se maximiza $NF$.
  }
\end{frame}
