\begin{frame}
  \begin{defn}
    Sea $f: U\subseteq\Real^n \to \Real$ una funciÃ³n. La derivada parcial de $f$ con respecto a la variable $x_i$ en el punto $a=(a_1, \ldots, a_n)$, denotada por $f_{x_i}(a)$, es
      $$f_{x_i}(a) = \lim_{h \to 0} \frac{f(a_1, \ldots, a_i + h, \ldots, a_n) - f(a_1, \ldots, a_n)}{h}.$$
  \end{defn}
  \onslide<2->{
    \begin{rem}
      \begin{itemize}
        \item La derivada parcial $f_{x_i}(x_1,\ldots,x_n)$ define una funciÃ³n con dominio en $\Real^n$ y recorrido en $\Real$.
        \item Las derivadas parciales de una funciÃ³n $f:U\subseteq\Real^n \to \Real$ tambiÃ©n se denotan por 
          $$\frac{\partial f}{\partial x_i}(a),\ \frac{\partial }{\partial x_i}f(a),\ D_i(f)(a),\ f_{x_i},\ D_i(f),\ \frac{\partial}{\partial x_i}f$$
      \end{itemize}
    \end{rem}
  }
\end{frame}

\begin{frame}
  \begin{eg}
    \begin{enumerate}[<+->]
      \item Sea $f(x,y) = x^2 + y^2$. Calcular $f_x(1,2)$ y $f_y(1,2)$.\\[0.25cm]
      \item Sea $f(x,y) = x^2\cos(x^2-y^2)$. Calcular $f_x(x,y)$, $f_y(x,y)$.\\[0.5cm]
    \end{enumerate}
  \end{eg}
\end{frame}

\begin{frame}
  TambiÃ©n podemos calcular derivadas parciales de orden superior. Por ejemplo las segundas derivadas parciales de una funciÃ³n $f$ de dos variables son.\\
  $$f_{xy} = \frac{\partial}{\partial y}\left(\frac{\partial f}{\partial x}\right) = \frac{\partial^2 f}{\partial y\partial x}$$
  $$f_{yx} = \frac{\partial}{\partial x}\left(\frac{\partial f}{\partial y}\right) = \frac{\partial^2 f}{\partial x\partial y}$$
  $$f_{xx} = \frac{\partial}{\partial x}\left(\frac{\partial f}{\partial x}\right) = \frac{\partial^2 f}{\partial x^2}$$
  $$f_{yy} = \frac{\partial}{\partial y}\left(\frac{\partial f}{\partial y}\right) = \frac{\partial^2 f}{\partial y^2}$$
\end{frame}

\begin{frame}
  \begin{eg}
    \begin{enumerate}[<+->]
      \item Sea $f(x,y) = x^2 + y^2$. Calcular $f_{xy}(1,2)$ y $f_{yx}(1,2)$.\\[0.25cm]
      \item Sea $f(x,y) = x^2\cos(x^2-y^2)$. Calcular $f_{xy}(x,y)$, $f_{yx}(x,y)$.\\[0.5cm]
    \end{enumerate}
  \end{eg}
\end{frame}

\begin{frame}
  \begin{thm}
    Si $f:U\subseteq\Real^2\To\Real$, $U$ abierto, y $f_{xy}$ y $f_{yx}$ son continuas en $U$, entonces
    $$f_{xy}(a,b)=f_{yx}(a,b)$$
    para $(a,b)\in U$.
  \end{thm}
\end{frame}

\begin{frame}
  \begin{defn}
    Sea $f:U\subseteq\Real^n\To\Real$ y $\vec{a}\in U$. Diremos que $f$ es \textbf{diferenciable en $\boldsymbol{\vec{a}}$} si se cumplen las siguientes condiciones
    \begin{enumerate}[<+->]
      \item Existen las derivadas parciales de $f$ en $\vec{a}$.\\
      \item $\displaystyle\lim_{\vec{h}\to\vec{0}}\dfrac{\abs{f(\vec{a}+\vec{h})-f(\vec{a})-\nabla f(\vec{a})\cdot\vec{h}}}{\norm{\vec{h}}}=0$\\
    \end{enumerate}
    \onslide<3->{
    donde $\nabla f(\vec{a})=(f_{x_1}(\vec{a}),f_{x_2}(\vec{a}),\ldots,f_{x_n}(\vec{a}))$, $\vec{h}=(h_1,h_2,\ldots,h_n)$ y 
    $$\nabla f(\vec{a})\cdot\vec{h}=\sum_{i=1}^{n}f_{x_i}(\vec{a})h_i$$
    }
  \end{defn}
\end{frame}

\begin{frame}
  \begin{rem}
    El vector $\nabla f(\vec{x})$ se llama el \textbf{gradiente de $\boldsymbol{f}$ en $\boldsymbol{\vec{x}}$}, su versiÃ³n matricial
    $$Df(\vec{x})=\left[\begin{array}{rrrr}f_{x_1}(\vec{x}) & f_{x_2}(\vec{x}) & \cdots & f_{x_n}(\vec{x})\end{array}\right]$$
    se le llama la \textbf{derivada de $\boldsymbol{f}$ en $\boldsymbol{\vec{x}}$}. %o \textbf{derivada matricial de $f$ en $\vec{x}$}.
  \end{rem}
\end{frame}

\begin{frame}
  En $\Real^2$ el lÃ­mite de la diferenciabilidad de una funciÃ³n $f(x,y)$ en un punto $(a,b)$ se puede escribir como
  $$\lim_{(x,y)\to(a,b)}\dfrac{\abs{f(x,y)-f(a,b)-f_x(a,b)h-f_y(a,b)k}}{\sqrt{h^2+k^2}}=0$$
  donde $h=x-a$ y $k=y-b$.
\end{frame}

\begin{frame}
  La funciÃ³n
  $$L(x,y)=f(a,b)+f_x(a,b)(x-a)+f_y(a,b)(y-b)$$
  se llama \textbf{aproximaciÃ³n lineal de $\boldsymbol{f}$ en $\boldsymbol{(a,b)}$} o \textbf{plano tangente a la grÃ¡fica de $\boldsymbol{f}$ en $\boldsymbol{(a,b)}$}, que tambiÃ©n se puede escribir como
  $$z=f(a,b)+f_x(a,b)(x-a)+f_y(a,b)(y-b)$$
  o
  $$f_x(a,b)(x-a)+f_y(a,b)(y-b)-z+f(a,b)=0$$
\end{frame}

\begin{frame}
  \begin{eg}
    El plano tangente a la grÃ¡fica de la funciÃ³n $f(x,y)=x^2+y^2$ en el punto $(2,2)$ es
    \onslide<2->{
    $$z=4(x-2)+4(y-2)+8$$
    o
    $$4x+4y-z-8=0$$
    }
  \end{eg}
\end{frame}

\begin{frame}
  \begin{thm}
    Sea $f:U\subseteq\Real^n\To\Real$ y $\vec{a}\in U$. Si $f$ es diferenciable en $\vec{a}$, entonces $f$ es continua en $\vec{a}$.
  \end{thm}
  \onslide<2->{
    \begin{thm}
      Sea $f:U\subseteq\Real^n\To\Real$, con $U$ abierto. Si sus derivadas parciales son continuas en $U$, entonces $f$ es diferenciable en todo punto de $U$.
    \end{thm}
  }
\end{frame}

\begin{frame}
  \begin{defn}
    Sea $f:U\subseteq\Real^n\To\Real$ y $\vec{a}\in U$. La \textbf{derivada direccional de $f$ en $\vec{a}$ en la direcciÃ³n de $\vec{v}$}, $\vec{v}$ un vector unitario, es
    $$D_{\vec{v}}f(\vec{a})=\lim_{h\to0}\dfrac{f(\vec{a}+h\vec{v})-f(\vec{a})}{h}$$
  \end{defn}
\end{frame}

\begin{frame}
  \begin{thm}
    Sea $f:U\subseteq\Real^n\To\Real$ y $\vec{a}\in U$. Si $f$ es diferenciable en $\vec{a}$, entonces
    $$D_{\vec{v}}f(\vec{a})=\nabla f(\vec{a})\cdot\vec{v}$$
    donde $\vec{v}$ es un vector unitario.
  \end{thm}
  \onslide<2->{
    \begin{rem}
      No siempre se cumple que $\vec{v}$ sea un vector unitario, en cuyo caso se calcula el vector unitario en la direcciÃ³n de $\vec{v}$ como
      $$\hat{v}=\dfrac{\vec{v}}{\norm{v}}$$
    \end{rem}
  }
\end{frame}

\begin{frame}
  \begin{eg}
    La derivada direccional de la funciÃ³n $f(x,y)=x^2+y^2$ en el punto $(2,2)$ en la direcciÃ³n del vector $\vec{v}=(1,1)$ es
    \onslide<2->{
    \begin{eqnarray*}
      D_{\hat{v}}f(2,2) &=& \nabla f(2,2)\cdot\frac{\vec{v}}{\norm{v}}\\
      \onslide<3->{
      &=& (4,4)\cdot\frac{(1,1)}{\sqrt{2}}\\
      % &=& \frac{8}{\sqrt{2}}\\
      &=& 4\sqrt{2}
      }
    \end{eqnarray*}
    }
  \end{eg}
\end{frame}

\begin{frame}
  \begin{thm}
    Sea $f:U\subseteq\Real^n\To\Real$, $f$ diferenciable en $U$ y $\vec{a}\in U$. Si $\nabla f(\vec{a})\neq\vec{0}$, y $f(\vec{a})=c$, entonces $\nabla f(\vec{a})$ es ortogonal al conjunto de nivel $f(\vec{x})=c$ en $\vec{a}$.
  \end{thm}
\end{frame}

\begin{frame}
  \begin{eg}
    Sea $f(x,y)=x^2y+xy\cos(y)$. La recta tangente a la curva de nivel $f(x,y)=0$ en el punto $(1,0)$ es
    \onslide<2->{
    \begin{eqnarray*}
      \nabla f(1,0) \cdot(x-1,y-0) &=& 0\\
      \onslide<3->{
      (0,2)\cdot(x-1,y) &=& 0\\
      0(x-1)+2(y) &=& 0\\
      y &=& 0
      }
    \end{eqnarray*}
    }
  \end{eg}
\end{frame}

\begin{frame}
  \begin{thm}
    Sea $f:U\subseteq\Real^n\to\Real$ y $\vec{a}\in U$. Si $f$ es diferenciable en $\vec{a}$, entonces la direcciÃ³n de mÃ¡ximo crecimiento de $f$ en $\vec{a}$ es $\nabla f(\vec{a})$ y el mÃ¡ximo valor de crecimiento es $\norm{\nabla f(\vec{a})}$.
  \end{thm}
\end{frame}

\begin{frame}
  \begin{defn}
    Sea $U\subseteq\Real^n$ abierto, $F:U\To\Real^m$ y $\vec{a}\in\Real^n$ un punto de acumulaciÃ³n de $U$. Diremos que $\vec{L}\in\Real^m$ es el \textbf{lÃ­mite de $F$ cuando $\vec{x}$ tiende a $\vec{a}$}, si y solo si
    $$\forall\epsilon>0\ \exists\delta>0\ \forall\vec{x}\in U\ \norm{\vec{x}-\vec{a}}<\delta\Rightarrow\norm{F(\vec{x})-\vec{L}}<\epsilon$$
    y lo denotaremos
    $$\lim_{\vec{x}\to\vec{a}}F(\vec{x})=\vec{L}$$
  \end{defn}
\end{frame}

\begin{frame}
  \begin{thm}
    Sea $U\subseteq\Real^n$ abierto, $F:U\To\Real^m$ y $\vec{a}\in\Real^n$ un punto de acumulaciÃ³n de $U$. Si $F=(f_1,f_2,\ldots,f_m)$, entonces
    $$\lim_{\vec{x}\to\vec{a}}F(\vec{x})=\vec{L}\Longleftrightarrow\lim_{\vec{x}\to\vec{a}}f_i(\vec{x})=L_i,\ i=1,2,\ldots,m$$
    donde $\vec{L}=(L_1,L_2,\ldots,L_m)$
  \end{thm}
\end{frame}

\begin{frame}
  \begin{defn}
    Sea $F:U\subseteq\Real^n\To\Real^m$ y $\vec{a}\in U$. Diremos que $f$ es \textbf{continua en $\vec{a}$} si y solo si
    $$\lim_{\vec{x}\to\vec{a}}F(\vec{x})=f(\vec{a})$$
  \end{defn}
  \onslide<2->{
    \begin{thm}
      Sea $F:U\subseteq\Real^n\To\Real^m$ y $\vec{a}\in U$, si $F=(f_1,f_2,\ldots,f_m)$, entonces $F$ es continua en $\vec{a}$ si y solo si $f_i$ es continua en $\vec{a}$, $i=1,2,\ldots,m$.
    \end{thm}
  }
\end{frame}

\begin{frame}
  \begin{defn}
    Sea $F:U\subseteq\Real^n\To\Real^m$ y $\vec{a}\in U$. Diremos que $F$ es \textbf{diferenciable en $\vec{a}$} si y solo si existe una matriz $A\in\mathcal{M}_{m\times n}(\Real)$ tal que
    $$\lim_{\vec{h}\to\vec{0}}\dfrac{\norm{F(\vec{a}+\vec{h})-F(\vec{a})-A\vec{h}}}{\norm{\vec{h}}}=0$$
    y denotaremos
    $$D(F)(\vec{a})=A$$
  \end{defn}
  \onslide<2->{
    \begin{thm}
      Sea $F:U\subseteq\Real^n\To\Real^m$ y $\vec{a}\in U$. Si $F=(f_1,f_2,\ldots,f_m)$, entonces $F$ es diferenciable en $\vec{a}$ si y solo si $f_i$ es diferenciable en $\vec{a}$, $i=1,2,\ldots,m$.
    \end{thm}
  }
\end{frame}

\begin{frame}
  \begin{rem}
    Si $F:U\subseteq\Real^n\To\Real^m$ y $\vec{a}\in U$, entonces $D(F)(\vec{a})$ es la matriz jacobiana de $F$ en $\vec{a}$.
    $$D(F)(\vec{a})=\begin{pmatrix}
      \dfrac{\partial f_1}{\partial x_1}(\vec{a}) & \dfrac{\partial f_1}{\partial x_2}(\vec{a}) & \cdots & \dfrac{\partial f_1}{\partial x_n}(\vec{a})\\
      \dfrac{\partial f_2}{\partial x_1}(\vec{a}) & \dfrac{\partial f_2}{\partial x_2}(\vec{a}) & \cdots & \dfrac{\partial f_2}{\partial x_n}(\vec{a})\\
      \vdots & \vdots & \ddots & \vdots\\
      \dfrac{\partial f_m}{\partial x_1}(\vec{a}) & \dfrac{\partial f_m}{\partial x_2}(\vec{a}) & \cdots & \dfrac{\partial f_m}{\partial x_n}(\vec{a})\end{pmatrix}=
      \begin{pmatrix}
        \nabla f_1(\vec{a})\\
        \nabla f_2(\vec{a})\\
        \vdots\\
        \nabla f_m(\vec{a})
      \end{pmatrix}
    $$
    que tambiÃ©n se denota por
    $$D(F)(\vec{a})=D_{\vec{x}}(F)(\vec{a})=D_{(x_1,x_2,\ldots,x_n)}(F)(\vec{a})$$
  \end{rem}
\end{frame}

\begin{frame}
  \begin{thm}
    Sean $f:U\subseteq\Real^2\To\Real$ y $g:V\subseteq\Real\To\Real^2$, funciones diferenciables, digamos $g(t)=(x(t),y(t))$, entonces
    \begin{eqnarray*}
      \dfrac{d}{dt}f(x(t),y(t)) &=& f_x(x(t),y(t))x'(t)+f_y(x(t),y(t))y'(t)
    \end{eqnarray*}
    \onslide<2->{
    en notaciÃ³n de Leibniz
    $$\dfrac{df}{dt}=\dfrac{\partial f}{\partial x}\dfrac{dx}{dt}+\dfrac{\partial f}{\partial y}\dfrac{dy}{dt}$$
    }
  \end{thm}
\end{frame}

\begin{frame}
  \begin{egs}
    \begin{enumerate}[<+->]
      \item Sea $f(x,y)=x^2+y^2$ y $g(t)=(\cos(t),\sen(t))$. Calcule $\dfrac{df}{dt}$.\\[0.5cm]
      \item Sea $f(x,y)=x^2\cos(x^2-y^2)$ y $g(t)=(t,t^2)$. Calcule $\dfrac{df}{dt}$.
    \end{enumerate}
  \end{egs}
\end{frame}

\begin{frame}
  \begin{thm}
    Sean $f:U\subseteq\Real^2\To\Real$ y $g:V\subseteq\Real^2\To\Real^2$, funciones diferenciables, digamos $g(u,v)=(x(u,v),y(u,v))$, entonces
    \begin{eqnarray*}
      \dfrac{\partial}{\partial u}f(x,y) &=& f_x(x,y)x_u(u,v)+f_y(x,y)y_u(u,v)\\
      \dfrac{\partial}{\partial v}f(x,y) &=& f_x(x,y)x_v(u,v)+f_y(x,y)y_v(u,v)
    \end{eqnarray*}
    \onslide<2->{
