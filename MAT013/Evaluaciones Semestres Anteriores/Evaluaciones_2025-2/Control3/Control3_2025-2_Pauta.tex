%! Pregunta 1
\noindent\textbf{P1) } 
  \begin{enumerate}[a)]
    \item Encuentre y clasifique los puntos críticos de
        $$f(x,y) = 8x^3-12xy+y^3$$
      % {\emph Ayuda:} $x^3-1=(x-1)(x^2+x+1)$
    \item Determine los máximos y mínimos globales de 
        $$g(x,y) = x^2-y^2$$
      en el conjunto $D = \set{(x,y) \in \Real^2 : \dfrac{x^2}{4}+y^2 = 1}$.
  \end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Para encontrar los puntos críticos de $f$ calcularemos las derivadas parciales
      \begin{eqnarray*}
        f_x(x,y) &=& 24x^2 - 12y \\
        f_y(x,y) &=& -12x + 3y^2
      \end{eqnarray*}
    Igualando a cero, obtenemos el sistema de ecuaciones
      \begin{eqnarray*}
        24x^2 - 12y &=& 0 \\
        -12x + 3y^2 &=& 0
      \end{eqnarray*}
    Cuyas soluciones son $(0,0)$ y $(1,2)$. Para clasificar los puntos críticos, usaremos el discriminante
      $$D(x,y) = f_{xx}(x,y)f_{yy}(x,y) - (f_{xy}(x,y))^2$$
    donde las segundas derivadas son
      \begin{eqnarray*}
        f_{xx}(x,y) &=& 48x \\
        f_{yy}(x,y) &=& 6y \\
        f_{xy}(x,y) &=& -12.
      \end{eqnarray*}
    Evaluando en los puntos críticos
    \begin{enumerate}[i)]
      \item En $(0,0)$
        \begin{eqnarray*}
          D(0,0) &=& f_{xx}(0,0)f_{yy}(0,0) - (f_{xy}(0,0))^2 \\
            &=& 48\cdot 0\cdot 6\cdot 0 - (-12)^2 \\
            &=& -144 < 0.
        \end{eqnarray*}
        Luego en el punto $(0,0)$ hay un punto silla.
      \item En $(1,2)$
        \begin{eqnarray*}
          D(1,2) &=& f_{xx}(1,2)f_{yy}(1,2) - (f_{xy}(1,2))^2 \\
            &=& 48\cdot 1\cdot 6\cdot 2 - (-12)^2 \\
            &=& 576 - 144 \\
            &=& 432 > 0.
        \end{eqnarray*}
        Además, $f_{xx}(1,2) = 48 > 0$, por lo que en el punto $(1,2)$ hay un mínimo local.
    \end{enumerate}
  % \newpage
  \item Para determinar los máximos y mínimos globales de $f$ en el conjunto $D = \set{(x,y) \in \Real^2 : \dfrac{x^2}{4}+y^2 = 1}$, usaremos el método de los multiplicadores de Lagrange. Definimos la función
    $$L(x,y,\lambda) = g(x,y) - \lambda \left(\dfrac{x^2}{4}+y^2-1\right)$$
    y calculamos las derivadas parciales
    \begin{eqnarray*}
      L_x(x,y,\lambda) &=& 2x - \lambda\cdot \frac{x}{2} \\
      L_y(x,y,\lambda) &=& -2y - 2\lambda y \\
      L_\lambda(x,y,\lambda) &=& -\left(\dfrac{x^2}{4}+y^2-1\right)
    \end{eqnarray*}
    Igualando a cero, obtenemos el sistema de ecuaciones
    \begin{eqnarray*}
      2x - \lambda\cdot \frac{x}{2} &=& 0 \\
      -2y - 2\lambda y &=& 0 \\
      -\left(\dfrac{x^2}{4}+y^2-1\right) &=& 0
    \end{eqnarray*}
    Cuyas soluciones son $(2,0)$, $(-2,0)$, $(0,1)$ y $(0,-1)$. Evaluando $g$ en estos puntos, obtenemos
    \begin{eqnarray*}
      g(2,0) &=& 2^2 - 0^2 = 4 \\
      g(-2,0) &=& (-2)^2 - 0^2 = 4 \\
      g(0,1) &=& 0^2 - 1^2 = -1 \\
      g(0,-1) &=& 0^2 - (-1)^2 = -1
    \end{eqnarray*}
    Por lo tanto, el máximo global de $g$ es $4$ y ocurre en los puntos $(2,0)$ y $(-2,0)$, mientras que el mínimos global es $-1$ y ocurre en los puntos $(0,1)$ y $(0,-1)$.
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) }
\begin{enumerate}[a)]
  \item En una piscina rectangular de 6 metros de largo, 4 metros de ancho, se toman a intervalos regulares la profundidad del agua. Los datos obtenidos son los siguientes.
    \begin{center}
    \begin{tabular}{|c||r|r|r|r|}
      \hline
          & 0 &  2 &  4 &  6 \\
      \hline
      \hline
      0   &   1 & 1.3 & 1.6 & 1.9 \\
      \hline
      2   & 1.1 & 1.2 & 1.7 &   2 \\
      \hline
      4   &   1 & 1.2 & 1.8 &   2 \\
      \hline
    \end{tabular}
    \end{center}
    estime el volumen de agua de la piscina usando sumas de Riemann, con el largo subdividido en 3 intervalos y el ancho en 2 intervalos.
  \item calcule
    $$\iint\limits_{R}\ e^{x+3y}\ dA$$
    donde $R=\set{(x,y)\in\Real^2:\ 0\leq x\leq 1 \wedge 0\leq y\leq 1}$.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Ordenaremos los datos como en la figura.

  \vspace{-5mm}
  \begin{figure}[h]
    \begin{center}
    \includegraphics[scale=0.6]{./imagenes/Pregunta2a.png}
    \end{center}
  \end{figure}

  \vspace{-5mm}
  A los elementos del largo los llamaremos $x_i$ y a los elementos del ancho los llamaremos $y_j$. Luego
  $$x_0=0,\ x_1=2,\ x_2=4,\ x_3=6$$
  $$y_0=0,\ y_1=2,\ y_2=4$$
  y los valores de $f(x_i,y_j)$, son los datos de la tabla. Luego, tomando los extremos superior derecho de cada rectángulo, tenemos que el volumen aproximado de la piscina es
  \begin{eqnarray*}
    V &\approx& \sum_{i=1}^3\sum_{j=1}^2f(x_i,y_j)\Delta x\Delta y\\
    V &\approx& f(2,2)\cdot2\cdot2+f(4,2)\cdot2\cdot2+f(6,2)\cdot2\cdot2+f(2,4)\cdot2\cdot2+f(4,4)\cdot2\cdot2+f(6,4)\cdot2\cdot2\\
    V &\approx& 1.2\cdot2\cdot2+1.7\cdot2\cdot2+2\cdot2\cdot2+1.2\cdot2\cdot2+1.8\cdot2\cdot2+2\cdot2\cdot2\\
    V &\approx& 4.8+6.8+8+4.8+7.2+8\\
    V &\approx& 39.6
  \end{eqnarray*}
  \item 
    \begin{eqnarray*}
      \iint\limits_{R}\ e^{x+3y}\ dA
        &=& \int_0^1\int_0^1 e^{x+3y}\ dydx\\
        &=& \int_0^1\int_0^1 e^x\cdot e^{3y}\ dydx\\
        &=& \int_0^1 e^x\left(\int_0^1 e^{3y}\ dy\right)dx\\
        &=& \int_0^1 e^x\left[\frac{e^{3y}}{3}\right]_{0}^{1} dx\\
        &=& \int_0^1 e^x\left(\frac{e^{3}-1}{3}\right) dx\\
        &=& \frac{e^{3}-1}{3}\int_0^1 e^x\ dx\\
        &=& \frac{e^{3}-1}{3}\left[e^x\right]_{0}^{1}\\
        &=& \frac{e^{3}-1}{3}(e-1)\\
        &=& \frac{(e^{3}-1)(e-1)}{3}\\
        &=& \frac{e^{4}-e^{3}-e+1}{3}
    \end{eqnarray*}
\end{enumerate}
