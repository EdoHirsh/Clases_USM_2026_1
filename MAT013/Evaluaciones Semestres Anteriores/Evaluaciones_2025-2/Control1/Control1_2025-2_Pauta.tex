%! Pregunta 1
\noindent\textbf{P1) } Calcule los siguientes límites.
\begin{enumerate}[a)]
  \item $\displaystyle\lim_{n\to\infty}\dfrac{\sqrt{4n^4+1}}{n^2+1}$
  \item $\displaystyle\lim_{n\to\infty}\ln\left(\dfrac{2n^2+2n-1}{n^2-n+1}\right)$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      \lim_{n\to\infty}\dfrac{\sqrt{4n^4+1}}{n^2+1}
        &=& \lim_{n\to\infty}\dfrac{\sqrt{4n^4+1}}{n^2+1}\cdot\frac{\frac{1}{n^2}}{\frac{1}{n^2}}\\
        &=& \lim_{n\to\infty}\dfrac{\sqrt{4+\frac{1}{n^4}}}{1+\frac{1}{n^2}}\\
        &=& \dfrac{\sqrt{4}}{1}\\
        &=& 2
    \end{eqnarray*}
  \item
    \begin{eqnarray*}
      \lim_{n\to\infty}\ln\left(\dfrac{2n^2+2n-1}{n^2-n+1}\right)
        &=& \ln\left(\lim_{n\to\infty}\dfrac{2n^2+2n-1}{n^2-n+1}\right)\\
        &=& \ln\left(\lim_{n\to\infty}\dfrac{2n^2+2n-1}{n^2-n+1}\cdot\frac{\frac{1}{n^2}}{\frac{1}{n^2}}\right)\\
        &=& \ln\left(\lim_{n\to\infty}\dfrac{2+\frac{2}{n}-\frac{1}{n^2}}{1-\frac{1}{n}+\frac{1}{n^2}}\right)\\
        &=& \ln(2)
    \end{eqnarray*}
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) } Calcule el valor de la serie, si es posible, si no justifique.
\begin{enumerate}[a)]
  \item $\displaystyle\sum_{n=0}^{\infty}\dfrac{6}{3^{n+1}}$
  \item $\displaystyle\sum_{n=1}^{\infty}\dfrac{n-1}{\sqrt{n^2+1}}$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      \sum_{n=0}^{\infty}\dfrac{6}{3^{n+1}}
        &=& 2\sum_{n=0}^{\infty}\left(\dfrac{3}{3^{n+1}}\right)\\
        &=& 2\sum_{n=0}^{\infty}\left(\dfrac{1}{3^n}\right)\\
        &=& 2\cdot\frac{1}{1-\frac{1}{3}}\\
        &=& 2\cdot\frac{3}{2}\\
        &=& 3
    \end{eqnarray*}
  \item la serie diverge, porque
    \begin{eqnarray*}
      \lim_{n\to\infty}\dfrac{n-1}{\sqrt{n^2+1}}
        &=& \lim_{n\to\infty}\dfrac{n-1}{\sqrt{n^2+1}}\cdot\frac{\frac{1}{n}}{\frac{1}{n}}\\
        &=& \lim_{n\to\infty}\dfrac{1-\frac{1}{n}}{\sqrt{1+\frac{1}{n^2}}}\\
        &=& 1
    \end{eqnarray*}
    que es distinto de cero.
\end{enumerate}
