  %* Pregunta 1
  \item \textsf{[40 pts.]} Sea
    $$f(x,y) = x^2+y^2+4x-4y$$
    \begin{enumerate}
      \item Calcule las primeras y segundas derivadas parciales de $f$.
      \vspace{2mm}
      \item Encuentre y clasifique los puntos críticos de $f$.
      \vspace{2mm}
      \item Encuentre los valores máximos y mínimos absolutos de $f$ en la circunferencia $x^2+y^2 = 9$.
    \end{enumerate}

  \vspace{5mm}
  %* Pregunta 2
  \item \textsf{[30 pts.]} Calcule las siguientes integrales dobles.
    \begin{enumerate}[a)]
      \item $\displaystyle\iint_D\ y^2\ dA$, donde $D$ es el triangulo con vértices en $(0,1)$, $(1,2)$ y $(4,1)$. 
      \item $\displaystyle\iint_D\ \dfrac{y^2}{x^2+y^2}\ dA$, donde $D$ la región entre las circunferencias $x^2+y^2=1$ y $x^2+y^2=4$. 
    \end{enumerate}

  \vspace{5mm}
  %* Pregunta 3
  \item \textsf{[30 pts.]} Calcule las siguientes integrales triples.
    \begin{enumerate}[a)]
      \item $\displaystyle\iiint_D\ x^2-xy+z^2x\ dV$, con $D=[0,2]\times[0,1]\times[0,3]$. 
      \item $\displaystyle\iiint_D\ 6xy\ dV$, con $D=\set{(x,y,z):0\leq y\leq 1\wedge 0\leq x \leq y^2 \wedge 0\leq z \leq 1+x+y}$. 
    \end{enumerate}
\end{enumerate}

{\large Integrales básicas:}
\begin{multicols}{2}
  \begin{enumerate}
    \item $\displaystyle\int x^n dx = \frac{x^{n+1}}{n+1}+C$ (si $n\neq -1$).
    \item $\displaystyle\int \frac{1}{x}dx = \ln|x|+C$
    \item $\displaystyle\int e^x dx = e^x+C$
    \item $\displaystyle\int \ln(x) dx = x\ln(x)-x+C$
    \item $\displaystyle\int \sin(x) dx = -\cos x+C$
    \item $\displaystyle\int \cos(x) dx = \sin x+C$
    \item $\displaystyle\int \sec^2(x) dx = \tan x+C$
    \item $\displaystyle\int \csc^2(x) dx = -\cot x+C$
    \item $\displaystyle\int \frac{1}{1+x^2}dx = \arctan x+C$
    \item $\displaystyle\int \frac{1}{\sqrt{1-x^2}}dx = \arcsin x+C$
    \item $\displaystyle\int \sec(x) dx = \ln|\sec x+\tan x|+C$
    \item $\displaystyle\int \csc(x) dx = \ln|\csc x-\cot x|+C$
    \item $\displaystyle\int \tan(x) dx = -\ln|\cos x|+C$
    \item $\displaystyle\int \cot(x) dx = \ln|\sin x|+C$
    \item $\displaystyle\int \cosh(x) dx = \sinh x+C$
    \item $\displaystyle\int \sinh(x) dx = \cosh x+C$
  \end{enumerate}
\end{multicols}
\vspace{2mm}

{\large Derivadas básicas:}
\begin{multicols}{2}
  \begin{enumerate}
    \item $(x^n)' = nx^{n-1}$
    \item $(e^x)' = e^x$
    \item $(\ln\abs{x})' = \frac{1}{x}$
    \item $(\sen(x))' = \cos(x)$
    \item $(\cos(x))' = -\sen(x)$
    \item $(\tan(x))' = \sec^2(x)$
    \item $(\cot(x))' = -\csc^2(x)$
    \item $(\sec(x))' = \sec(x)\tan(x)$
    \item $(\csc(x))' = -\csc(x)\cot(x)$
    \item $(\arcsen(x))' = \frac{1}{\sqrt{1-x^2}}$
    \item $(\arccos(x))' = -\frac{1}{\sqrt{1-x^2}}$
    \item $(\arctan(x))' = \frac{1}{1+x^2}$
    \item $(\cosh(x))' = \sinh(x)$
    \item $(\sinh(x))' = \cosh(x)$
  \end{enumerate}
\end{multicols}
