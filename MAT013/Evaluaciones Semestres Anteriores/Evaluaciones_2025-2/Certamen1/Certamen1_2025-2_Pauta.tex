%! Pregunta 1
\noindent\textbf{P1) } \textsf{[ 30 Pts. ] } Calcule.
\begin{enumerate}[a)]
  \item $\displaystyle\lim_{n\to\infty}\sqrt{\dfrac{3n^3-n^2+2}{12n^3-n-1}}$.
  \item $\displaystyle\sum_{n=1}^{\infty} \frac{2\cdot 3^{n-1}}{(-6)^n}$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      \lim_{n\to\infty}\sqrt{\dfrac{3n^3-n^2+2}{12n^3-n-1}} 
        &=& \sqrt{\lim_{n\to\infty}\dfrac{3n^3-n^2+2}{12n^3-n-1}} \\
        % &=& \sqrt{\lim_{n\to\infty}\dfrac{3-\dfrac{n^2}{n^3}+\dfrac{2}{n^3}}{12-\dfrac{n}{n^3}-\dfrac{1}{n^3}}} \\
        &=& \sqrt{\lim_{n\to\infty}\dfrac{3-\dfrac{1}{n}+\dfrac{2}{n^3}}{12-\dfrac{1}{n^2}-\dfrac{1}{n^3}}} \\
        &=& \sqrt{\dfrac{3}{12}}\\
        % &=& \sqrt{\dfrac{1}{4}}\\
        &=& \dfrac{1}{2}
    \end{eqnarray*}
  \item
    \begin{eqnarray*}
      \sum_{n=1}^{\infty} \frac{2\cdot 3^{n-1}}{(-6)^n} 
        &=& \sum_{n=1}^{\infty} \frac{2\cdot3^{n-1}}{2^n(-1)^n3^n}\\
        % &=& \sum_{n=1}^{\infty} \frac{1}{3(-1)^n 2^{n-1}}\\
        &=& -\frac{1}{3}\sum_{n=1}^{\infty} \frac{1}{(-1)^{n-1}2^{n-1}}\\
        &=& -\frac{1}{3}\sum_{n=1}^{\infty} \left(-\frac{1}{2}\right)^{n-1}\\
        &=& -\frac{1}{3}\sum_{k=0}^{\infty} \left(-\frac{1}{2}\right)^{k}\\
        &=& -\frac{1}{3}\left(\dfrac{1}{1+\frac{1}{2}}\right)\\
        % &=& -\frac{1}{3}\left(\dfrac{1}{\frac{3}{2}}\right)\\
        % &=& -\frac{1}{3}\cdot \frac{2}{3}\\
        &=& -\frac{2}{9}
    \end{eqnarray*}
\end{enumerate}


%! Pregunta 2
\newpage
\noindent\textbf{P2) } \textsf{[ 30 Pts. ] } Indique si las siguientes series convergen o divergen. Justifique su respuesta.
\begin{enumerate}[a)]
  \item $\displaystyle\sum_{n=2}^{\infty}\dfrac{(-1)^n}{n^2-n}$
  \item $\displaystyle\sum_{n=0}^{\infty}\dfrac{\sqrt{n^3+1}}{2^n}$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Para esta serie comprobaremos que converge absolutamente, es decir, comprobaremos que la serie
    $$\sum_{n=2}^{\infty}\abs{\dfrac{(-1)^n}{n^2-n}}=\sum_{n=2}^{\infty}\dfrac{1}{n^2-n}$$
    converge. Para esto usaremos el criterio de comparación en el límite con la serie $\displaystyle\sum_{n=2}^{\infty}\dfrac{1}{n^2}$, para lo cual calculamos el límite
      \begin{eqnarray*}
        \lim_{n\to\infty}\dfrac{\frac{1}{n^2-n}}{\frac{1}{n^2}} 
          &=& \lim_{n\to\infty}\dfrac{n^2}{n^2-n}\\
          &=& \lim_{n\to\infty}\dfrac{1}{1-\frac{1}{n}}\\
          &=& 1
      \end{eqnarray*}
    Por lo tanto la serie $\displaystyle\sum_{n=2}^{\infty}\dfrac{1}{n^2-n}$ converge, luego la serie original converge absolutamente y por ende converge.
  \item Para esta serie utilizaremos el criterio del cuociente, para lo cual calculamos
    \begin{eqnarray*}
      \lim_{n\to\infty}\abs{\dfrac{a_{n+1}}{a_n}}
        &=& \lim_{n\to\infty}\abs{\dfrac{\frac{\sqrt{(n+1)^3+1}}{2^{n+1}}}{\frac{\sqrt{n^3+1}}{2^n}}}\\
        &=& \lim_{n\to\infty}\abs{\dfrac{1}{2}\dfrac{\sqrt{(n+1)^3+1}}{\sqrt{n^3+1}}}\\
        &=& \lim_{n\to\infty}\abs{\dfrac{1}{2}\sqrt{\dfrac{n^3+3n^2+3n+2}{n^3+1}}}\\
        &=& \lim_{n\to\infty}\abs{\dfrac{1}{2}\sqrt{\dfrac{1+\frac{3}{n}+\frac{3}{n^2}+\frac{2}{n^3}}{1+\frac{1}{n^3}}}}\\
        &=& \dfrac{1}{2}
    \end{eqnarray*}
    Como es menor que $1$ la serie converge.
\end{enumerate}


%! Pregunta 3
\newpage
\noindent\textbf{P3) }\textsf{[ 40 Pts. ] }
\begin{enumerate}[a)]
  \item \textsf{[ 20 Pts. ] }Encuentre el intervalo de convergencia de
    $$\sum_{n=0}^{\infty}\dfrac{(-5x)^n}{n+1}$$
  \item \textsf{[ 10 Pts. ] }Encuentre la serie que representa la función $f(x)=(x-1)^2\sen(x-1)$ indicando su centro y radio de convergencia.
  \item \textsf{[ 10 Pts. ] }Use el polinomio de Taylor de orden 5 para aproximar $f(1,5)$, donde $f$ es la función del item anterior.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Usando el criterio del cuociente calculamos
    \begin{eqnarray*}
      \lim_{n\to\infty}\abs{\dfrac{a_{n+1}}{a_n}}
        &=& \lim_{n\to\infty}\abs{\dfrac{\frac{(-5x)^{n+1}}{n+2}}{\frac{(-5x)^n}{n+1}}}\\
        &=& \lim_{n\to\infty}\abs{\dfrac{(-5x)^{n+1}}{(-5x)^n}\cdot\dfrac{n+1}{n+2}}\\
        &=& \lim_{n\to\infty}\abs{(-5x)\cdot\dfrac{n+1}{n+2}}\\
        &=& \lim_{n\to\infty}\abs{(-5x)}\cdot\lim_{n\to\infty}\dfrac{n+1}{n+2}\\
        &=& \abs{(-5x)}\cdot 1\\
        &=& 5\abs{x}
    \end{eqnarray*}
    Luego la serie converge absolutamente si $5\abs{x}<1$, es decir, si $\abs{x}<\frac{1}{5}$. Por lo cual el radio de convergencia es $R=\frac{1}{5}$ y el potencial intervalo de convergencia es $\intervaloAA{-\frac{1}{5},\frac{1}{5}}$, falta comprobar los extremos.
      \begin{itemize}
        \item Para $x=-\frac{1}{5}$, la serie se convierte en $\sum_{n=0}^{\infty}\dfrac{(-1)^{n}}{n+1}$, que converge por el criterio de Leibniz.
        \item Para $x=\frac{1}{5}$, la serie se convierte en $\sum_{n=0}^{\infty}\dfrac{1}{n+1}$, que diverge, por criterio de comparación con la serie armónica.
      \end{itemize}
    Por lo tanto el intervalo de convergencia es $\intervaloCA{-\frac{1}{5},\frac{1}{5}}$.
  \newpage
  \item Usando como punto de partida la serie de Taylor de $\sen(x)$ centrada en $x=0$, que es
    $$\sen(x)=\sum_{n=0}^{\infty}(-1)^n\dfrac{x^{2n+1}}{(2n+1)!}$$
    podemos escribir
    \begin{eqnarray*}
      f(x) 
        &=& (x-1)^2\sen(x-1)\\
        &=& (x-1)^2\sum_{n=0}^{\infty}(-1)^n\dfrac{(x-1)^{2n+1}}{(2n+1)!}\\
        &=& \sum_{n=0}^{\infty}(-1)^n\dfrac{(x-1)^2(x-1)^{2n+1}}{(2n+1)!}\\
        &=& \sum_{n=0}^{\infty}(-1)^n\dfrac{(x-1)^{2n+3}}{(2n+1)!}
    \end{eqnarray*}
    Por lo tanto la serie que representa a $f$ es $\displaystyle\sum_{n=0}^{\infty}(-1)^n\dfrac{(x-1)^{2n+3}}{(2n+1)!}$, cuyo centro es $x=1$. El radio de convergencia es el mismo que el de la serie de $\sen(x)$, que es infinito.
  \item Para encontrar el polinomio de Taylor de orden 5, desarrollaremos los primeros términos de la serie encontrada en el item anterior.
    \begin{eqnarray*}
      f(x) 
        &=& \sum_{n=0}^{\infty}(-1)^n\dfrac{(x-1)^{2n+3}}{(2n+1)!}\\
        &=& \dfrac{(x-1)^3}{1!}-\dfrac{(x-1)^5}{3!}+\dfrac{(x-1)^7}{5!}-\cdots
    \end{eqnarray*}
    Por lo tanto el polinomio de Taylor de orden 5 es
    $$T_5(x)= (x-1)^3-\dfrac{(x-1)^5}{6}$$
    Finalmente, para aproximar $f(1,5)$, calculamos
    \begin{eqnarray*}
      T_5(1,5) 
        &=& (1,5-1)^3-\dfrac{(1,5-1)^5}{6}\\
        &=& (0,5)^3-\dfrac{(0,5)^5}{6}\\
        &=& \frac{1}{2^3}-\dfrac{1}{6\cdot 2^5}\\
        &=& \frac{1}{8}-\dfrac{1}{192}\\
        &=& \frac{24}{192}-\dfrac{1}{192}\\
        &=& \frac{23}{192}
    \end{eqnarray*}
    Por lo tanto $f(1,5)\approx \dfrac{23}{192}$.%\approx 0,11979
\end{enumerate}
