%! Pregunta 1
\noindent\textbf{P1) } \textsf{[ 30 Pts. ]}
\begin{enumerate}[a)]
  \item Dado el siguiente gráfico de las curvas de nivel de una función $f(x,y)$.\\
    \vspace{-10mm}
    \begin{figure}[h!]
      \begin{center}
      \includegraphics[scale=0.6]{imagenes/Pregunta1a.png}
      \end{center}
    \end{figure}
    \vspace{-13mm}
    \par identifique, justificando su respuesta, cuál de las siguientes opciones podría corresponder a la función $f(x,y)$.
    \begin{multicols}{4}
      \begin{enumerate}[i.]
        \item $f(x,y)=6-2x^2-y^2$
        \item $f(x,y)=2x^2+y^2$
        \item $f(x,y)=6-x^2+2y^2$
        \item $f(x,y)=x^2-2y^2$
      \end{enumerate}
    \end{multicols}
  \item Identifique y esboce el gráfico de la superficie de ecuación.
    $$ x^2 - 4y^2- 4x + 8y + 4z - 20 = 0 $$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Analizamos cada una de las opciones.
    \begin{enumerate}[i.]
      \item En es caso las curvas de nivel $f(x,y)=z$ corresponden a elipses con eje mayor en el eje $y$, que se van achicando a medida que $z$ aumenta, lo que coincide con lo presentado en el gráfico.
      \item En este caso las curvas de nivel $f(x,y)=z$ corresponden a elipses con eje mayor en el eje $y$, que se van agrandando a medida que $z$ aumenta, lo que no coincide con lo presentado en el gráfico.
      \item En este caso las curvas de nivel $f(x,y)=z$ corresponden a hipérbolas con eje real en el eje $x$, que se van achicando a medida que $z$ aumenta, lo que no coincide con lo presentado en el gráfico.
      \item En este caso las curvas de nivel $f(x,y)=z$ corresponden a hipérbolas con eje real en el eje $x$, que se van agrandando a medida que $z$ aumenta, lo que no coincide con lo presentado en el gráfico.
    \end{enumerate}
    Por lo tanto la opción correcta es la i.
  % \vspace{2mm}
  \newpage
  \item completando cuadrados y reordenando obtenemos
    \begin{eqnarray*}
      x^2 - 4y^2- 4x + 8y + 4z - 20                      &=& 0 \\
      (x^2 - 4x + 4) - 4(y^2 - 2y + 1) + 4z - 20 - 4 + 4 &=& 0 \\
      (x-2)^2 - 4(y-1)^2 + 4z - 20                       &=& 0 \\
      (x-2)^2 - 4(y-1)^2                                 &=& 20 - 4z\\
      -(x-2)^2 + 4(y-1)^2                                &=& 4(z-5)
    \end{eqnarray*}
    que corresponde a un paraboloide hiperbólico con vértice en el punto $(2,1,5)$, esto es, una silla de montar de la forma $-u^2+v^2=4w$, trasladada de manera que el vértice $(0,0,0)$ se traslada a $(2,1,5)$. Gráficamente se ve como sigue.
    \vspace{-10mm}
    \begin{center}
      \includegraphics[scale=1]{./imagenes/Pregunta1b.png}
    \end{center}
\end{enumerate}

\noindent\textbf{P2) } \textsf{[ 40 Pts. ]} Dada la función
  $$f(x,y) =\begin{cases}
    3x^2+2xy+y^2  & \textrm{ si }\ 4x^2+y^2\leq 4\\
    \dfrac{y^3\cos(\pi x)}{(x-1)^2+y^2}  & \textrm{ si }\ 4x^2+y^2>4
  \end{cases}$$
  Calcule.
  \begin{enumerate}[a)]
    \item \textsf{[10 Pts.]}$\displaystyle\lim_{(x,y)\to(0,0)}f(x,y)$
    \item \textsf{[10 Pts.]}$\displaystyle\lim_{(x,y)\to(0,1)}f(x,y)$
    \item \textsf{[20 Pts.]}$\displaystyle\lim_{(x,y)\to(1,0)}f(x,y)$
  \end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Como $(0,0)$ está en el interior de la región $4x^2+y^2 \leq 4$, el límite se calcula como sigue.
    \begin{eqnarray*}
      \lim_{(x,y)\to(0,0)}f(x,y)  
        &=& \lim_{(x,y)\to(0,0)}3x^2+2xy+y^2\\
        &=& 3\cdot 0^2 + 2\cdot 0\cdot 0 + 0^2\\
        &=& 0
    \end{eqnarray*}
  \item Como $(0,1)$ está en el interior de la región $4x^2+y^2 \leq 4$, el límite se calcula como sigue.
    \begin{eqnarray*}
      \lim_{(x,y)\to(0,1)}f(x,y)  
        &=& \lim_{(x,y)\to(0,1)}3x^2+2xy+y^2\\
        &=& 3\cdot 0^2 + 2\cdot 0\cdot 1 + 1^2\\
        &=& 1
    \end{eqnarray*}
  % \newpage
  \item En este caso el punto $(1,0)$ está en la frontera entre las dos regiones, por lo que se debe analizar el límite por separado en cada región.
    \begin{enumerate}[i.]
      \item Si $(x,y)$ se acerca a $(1,0)$ por la región $4x^2+y^2\leq 4$, el límite se calcula como sigue.
        \begin{eqnarray*}
          % \lim_{\substack{(x,y)\to(1,0)\\4x^2+y^2\leq 4}}f(x,y)
          \lim_{(x,y)\to(1,0)}f(x,y)
            &=& \lim_{(x,y)\to(1,0)}3x^2+2xy+y^2\\
            &=& 3\cdot 1^2 + 2\cdot 1\cdot 0 + 0^2\\
            &=& 3
        \end{eqnarray*}
      \newpage
      \item Si $(x,y)$ se acerca a $(1,0)$ por la región $4x^2+y^2 > 4$, el límite se calcula como sigue.
        $$\lim_{(x,y)\to(1,0)}f(x,y) = \lim_{(x,y)\to(1,0)}\dfrac{y^3\cos(\pi x)}{(x-1)^2+y^2}$$
        En este caso, si se acerca por la línea $y=0$, se tiene que
        \begin{eqnarray*}
          \lim_{(x,y)\to(1,0)}f(x,y)
            &=& \lim_{x\to 1}\dfrac{0^3\cos(\pi x)}{(x-1)^2+0^2}\\
            &=& \lim_{x\to 1}\dfrac{0}{(x-1)^2}\\
            &=& 0
        \end{eqnarray*}
        luego en este caso podemos concluir que $\displaystyle\lim_{(x,y)\to(1,0)}f(x,y)\neq 3$.
    \end{enumerate}
    Por lo tanto el límite $\displaystyle\lim_{(x,y)\to(1,0)}f(x,y)$ no existe.
\end{enumerate}


%! Pregunta 3
\newpage
\noindent\textbf{P3) } \textsf{[ 30 Pts. ]} Dada la función
    $$f(x,y,z) = x^2+2xy-y^2z-z^2$$
  \begin{enumerate}[a)]
    \item Calcule las derivadas parciales de $f$.
    \item Encuentre la dirección de máximo crecimiento de $f$ en el punto $(1,1,2)$.
    \item Encuentre la ecuación del plano tangente a la superficie de ecuación
      $$f(x,y,z)=2$$
      en el punto $(2,1,2)$.
  \end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      f_x(x,y,z) &=& 2x + 2y\\
      f_y(x,y,z) &=& 2x - 2yz\\
      f_z(x,y,z) &=& -y^2 - 2z
    \end{eqnarray*}
  \item La dirección de máximo crecimiento de $f$ en el punto $(1,1,2)$ está dada por el gradiente de $f$ en ese punto. Este es.
    \begin{eqnarray*}
      \nabla f(1,1,2) 
        &=& (f_x(1,1,2), f_y(1,1,2), f_z(1,1,2))\\
        &=& (2\cdot 1 + 2\cdot 1, 2\cdot 1 - 2\cdot 1\cdot 2, -1^2 - 2\cdot 2)\\
        &=& (4, -2, -5)
    \end{eqnarray*}
    Por lo tanto la dirección de máximo crecimiento es en la dirección del vector $(4,-2,-5)$.
  \item Por teorema del gradiente, el vector normal al plano tangente a la superficie de nivel $f(x,y,z)=2$ en el punto $(2,1,2)$ está dado por el gradiente de $f$ en ese punto. Esto es.
    \begin{eqnarray*}
      \nabla f(2,1,2) 
        &=& (f_x(2,1,2), f_y(2,1,2), f_z(2,1,2))\\
        &=& (2\cdot 2 + 2\cdot 1, 2\cdot 2 - 2\cdot 1\cdot 2, -1^2 - 2\cdot 2)\\
        &=& (6, 0, -5)
    \end{eqnarray*}
    Por lo tanto la ecuación del plano tangente es.
    \begin{eqnarray*}
      6(x-2) + 0(y-1) - 5(z-2) &=& 0\\
      6x - 12 - 5z + 10 &=& 0\\
      6x - 5z - 2 &=& 0
    \end{eqnarray*}
\end{enumerate}
