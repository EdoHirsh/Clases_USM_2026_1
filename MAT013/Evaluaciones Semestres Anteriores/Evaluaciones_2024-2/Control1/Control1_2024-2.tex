\noindent\textbf{P1) } \textsf{[50 pts.]} Determine si las siguientes sucesiones convergen o divergen, y en caso de converger, encuentre su límite.
\begin{enumerate}[a)]
  \item $a_n = \dfrac{n^2-2}{\sqrt{n^2+1}}$
  \item $b_n = \cos\left(\dfrac{\pi n^3+n^2}{4n^3 + 3}\right)$
\end{enumerate}

Derivadas básicas:
\begin{multicols}{2}
  \begin{enumerate}
    \item $(x^n)' = nx^{n-1}$
    \item $(e^x)' = e^x$
    \item $(\ln\abs{x})' = \frac{1}{x}$
    \item $(\sen(x))' = \cos(x)$
    \item $(\cos(x))' = -\sen(x)$
    \item $(\tan(x))' = \sec^2(x)$
    \item $(\cot(x))' = -\csc^2(x)$
    \item $(\sec(x))' = \sec(x)\tan(x)$
    \item $(\csc(x))' = -\csc(x)\cot(x)$
    \item $(\arcsen(x))' = \frac{1}{\sqrt{1-x^2}}$
    \item $(\arccos(x))' = -\frac{1}{\sqrt{1-x^2}}$
    \item $(\arctan(x))' = \frac{1}{1+x^2}$
  \end{enumerate}
\end{multicols}

Identidades trigonométricas:
\begin{multicols}{2}
  \begin{enumerate}
    \item $\cos^2(x)+\sen^2(x)=1$
    \item $\tan^2(x)+1=\sec^2(x)$
    \item $\sen(x\pm y)=\sen(x)\cos(y)\pm\cos(x)\sen(y)$
    \item $\cos(x\pm y)=\cos(x)\cos(y)\mp\sen(x)\sen(y)$
    \item $\sen(2x)=2\sen(x)\cos(x)$
    \item $\cos(2x)=\cos^2(x)-\sen^2(x)$
    \item $\cos^2(x)=\frac{1+\cos(2x)}{2}$
    \item $\sen^2(x)=\frac{1-\cos(2x)}{2}$
  \end{enumerate}
\end{multicols}
