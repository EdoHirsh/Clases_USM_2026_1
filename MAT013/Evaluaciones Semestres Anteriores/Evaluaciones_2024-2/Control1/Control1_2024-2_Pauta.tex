%! Pregunta 1
\noindent\textbf{P1) } Determine si las siguientes sucesiones convergen o divergen, y en caso de converger, encuentre su límite.
\begin{enumerate}[a)]
  \item $a_n = \dfrac{n^2-2}{\sqrt{n^2+1}}$
  \item $b_n = \cos\left(\dfrac{\pi n^3+n^2}{4n^3 + 3}\right)$
\end{enumerate}

\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      \lim_{n\to\infty} a_n &=& \lim_{n\to\infty} \dfrac{n^2-2}{\sqrt{n^2+1}}\\
      &=& \lim_{n\to\infty} \dfrac{n^2-2}{\sqrt{n^2+1}}\cdot\dfrac{1/n}{1/n}\\
      &=& \lim_{n\to\infty} \dfrac{n-\frac{2}{n}}{\frac{\sqrt{n^2+1}}{n}}\\
      &=& \lim_{n\to\infty} \dfrac{n-\frac{2}{n}}{\sqrt{1+\frac{1}{n^2}}}\\
      &=& \infty
    \end{eqnarray*}
    Por lo tanto, la sucesión diverge a infinito.
  \item
    \begin{eqnarray*}
      \lim_{n\to\infty} b_n &=& \lim_{n\to\infty} \cos\left(\dfrac{\pi n^3+n^2}{4n^3 + 3}\right)\\
      &=& \cos\left(\lim_{n\to\infty}\dfrac{\pi n^3+n^2}{4n^3 + 3}\right)\\
      &=& \cos\left(\lim_{n\to\infty}\dfrac{\pi n^3+n^2}{4n^3 + 3}\dfrac{1/n^3}{1/n^3}\right)\\
      &=& \cos\left(\lim_{n\to\infty}\dfrac{\pi +\frac{1}{n}}{4 + \frac{3}{n^3}}\right)\\
      &=& \cos\left(\dfrac{\pi}{4}\right)\\
      &=& \dfrac{\sqrt{2}}{2}
    \end{eqnarray*}
    Por lo tanto, la sucesión converge a $\dfrac{\sqrt{2}}{2}$.
\end{enumerate}

%! Pregunta 2
\newpage
\noindent\textbf{P2) } Determine si la siguiente serie es convergente o divergente, en caso de ser convergente encuentre su suma y en caso de ser divergente justifique.
$$\sum_{n=1}^\infty \dfrac{3^{n-1}}{2\cdot5^n}$$

\noindent\textbf{Respuesta:} Reescribimos la serie como
\begin{eqnarray*}
  \sum_{n=1}^\infty \dfrac{3^{n-1}}{2\cdot5^n} &=& \sum_{n=1}^\infty \dfrac{3^{n-1}}{2\cdot5^n}\\
  &=& \sum_{n=1}^\infty \frac{3^{n-1}}{10\cdot5^{n-1}}\\
  &=& \sum_{n=1}^\infty \dfrac{1}{10}\left(\dfrac{3}{5}\right)^{n-1}\\
  &=& \dfrac{1}{10}\sum_{n=1}^\infty \left(\dfrac{3}{5}\right)^{n-1}\\
  &=& \dfrac{1}{10}\sum_{n=0}^\infty \left(\dfrac{3}{5}\right)^{n}\\
  &=& \dfrac{1}{10}\dfrac{1}{1-\frac{3}{5}}\\
  &=& \dfrac{1}{10}\dfrac{1}{\frac{2}{5}}\\
  &=& \dfrac{1}{10}\cdot\dfrac{5}{2}\\
  &=& \dfrac{1}{4}
\end{eqnarray*}
