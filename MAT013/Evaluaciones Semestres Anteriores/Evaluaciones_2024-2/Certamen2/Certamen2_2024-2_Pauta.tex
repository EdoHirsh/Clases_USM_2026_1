%! Pregunta 1
\noindent\textbf{P1) } 
\begin{enumerate}[a)]
  \item Encuentre polinomio de Taylor de grado 4 centrado en $x=0$ de la función $f(x)=xe^{2x}$
  \item Encuentre la matriz canónica de la transformación lineal $T:\Real^3\to\Real^3$ definida por
    $$T(x,y,z) = (x+2y+z, 2x+3y+2z, 3x+4y+3z)$$
\end{enumerate}

\vspace{0.25cm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Primero sabemos que la serie de Taylor de $e^x$ centrada en $x=0$ es
    $$e^x = \sum_{n=0}^{\infty}\dfrac{x^n}{n!}$$
    Ahora 
    \begin{eqnarray*}
      xe^{2x} &=& x\sum_{n=0}^{\infty}\dfrac{(2x)^n}{n!}\\
        &=& x\sum_{n=0}^{\infty}\dfrac{2^nx^n}{n!}\\
        &=& \sum_{n=0}^{\infty}\dfrac{2^nx^{n+1}}{n!}\\
        &=& \dfrac{2^0x^1}{0!} + \dfrac{2^1x^2}{1!} + \dfrac{2^2x^3}{2!} + \dfrac{2^3x^4}{3!} + \dfrac{2^4x^5}{4!} +\cdots\\
        &=& x + 2x^2 + 2x^3 + \dfrac{4}{3}x^4 + \dfrac{2}{3}x^5 + \cdots
    \end{eqnarray*}
    Por lo tanto el polinomio de Taylor de grado 4 centrado en $x=0$ de la función $f(x)=xe^{2x}$ es
    $$T_4(x) = x + 2x^2 + 2x^3 + \dfrac{4}{3}x^4$$

  \item Para encontrar la matriz canónica de la transformación lineal necesitamos calcular las imágenes de los vectores canónicos.
    \begin{eqnarray*}
      T(1,0,0) &=& (1,2,3)\\
      T(0,1,0) &=& (2,3,4)\\
      T(0,0,1) &=& (1,2,3)
    \end{eqnarray*}
    Por lo tanto la matriz canónica de la transformación lineal $T$ es
    $$\begin{bmatrix}
      1 & 2 & 1\\
      2 & 3 & 2\\
      3 & 4 & 3
    \end{bmatrix}$$
\end{enumerate}

%! Pregunta 3
\newpage
\noindent\textbf{P3) } Dada la función
$$f(x,y) =\begin{cases}
  \dfrac{x^2y}{x^2+y^2} & \text{si } x>y\\
  x^2+xy-y^4 & \text{si } x\leq y
\end{cases}$$
Calcule.
\begin{enumerate}[a)]
  \item \textsf{[10 Pts.] } $\displaystyle\lim_{(x,y)\to(1,2)}f(x,y)$
  \item \textsf{[10 Pts.] }$\displaystyle\lim_{(x,y)\to(2,1)}f(x,y)$
  \item \textsf{[20 Pts.] }$\displaystyle\lim_{(x,y)\to(0,0)}f(x,y)$
\end{enumerate}

\vspace{0.25cm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Como $(1,2)$ satisface $x<y$ entonces
    \begin{eqnarray*}
      \lim_{(x,y)\to(1,2)}f(x,y) &=& \lim_{(x,y)\to(1,2)} x^2+xy-y^4\\
        &=& 1^2+1\cdot 2 - 2^4\\
        &=& 1+2-16\\
        &=& -13
    \end{eqnarray*}
  \item Como $(2,1)$ satisface $x>y$ entonces
    \begin{eqnarray*}
      \lim_{(x,y)\to(2,1)}f(x,y) &=& \lim_{(x,y)\to(2,1)} \dfrac{x^2y}{x^2+y^2}\\
        &=& \dfrac{2^2\cdot 1}{2^2+1^2}\\
        % &=& \dfrac{4}{4+1}\\
        &=& \dfrac{4}{5}
    \end{eqnarray*}
  \item Como $(0,0)$ satisface $x=y$, entonces debemos separar en 2 casos
    \begin{enumerate}[i)]
      \item si $x>y$
        \begin{eqnarray*}
          \lim_{(x,y)\to(0,0)}f(x,y) &=& \lim_{(x,y)\to(0,0)} \dfrac{x^2y}{x^2+y^2}\\
        \end{eqnarray*}
        notemos que
          $$0 \leq \abs{\dfrac{x^2y}{x^2+y^2}}\leq \abs{\dfrac{x^2y}{x^2}} = \abs{y}$$
        luego como $\lim\limits_{(x,y)\to(0,0)}y=0$ por el teorema del sándwich
          $$\lim_{(x,y)\to(0,0)}f(x,y) = 0$$
      \item si $x\leq y$
        \begin{eqnarray*}
          \lim_{(x,y)\to(0,0)}f(x,y) &=& \lim_{(x,y)\to(0,0)} x^2+xy-y^4\\
            &=& 0^2+0\cdot 0 - 0^4\\
            &=& 0
        \end{eqnarray*}
    \end{enumerate}
    Por lo tanto
      $$\lim_{(x,y)\to(0,0)}f(x,y) = 0$$
\end{enumerate}
