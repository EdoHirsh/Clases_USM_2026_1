%! Pregunta 1
\noindent\textbf{P1) } 
\begin{enumerate}[a)]
  \item Encuentre la serie de Maclaurin de la función $f(x)=x\cos(2x)$.
  \item Use un polinomio de Taylor de grado 5 para aproximar el valor de $f(0.5)$.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Sabemos que la serie de $\cos(x)$ es
    $$\cos(x)=\sum_{n=0}^\infty \frac{(-1)^n}{(2n)!}x^{2n}$$
    Por lo tanto, la serie de Maclaurin de $f(x)=x\cos(2x)$ es
    \begin{eqnarray*}
      f(x) &=& x\cos(2x)\\
      &=& x\sum_{n=0}^\infty \frac{(-1)^n}{(2n)!}(2x)^{2n}\\
      &=& x\sum_{n=0}^\infty \frac{(-1)^n}{(2n)!}2^{2n}x^{2n}\\
      &=& \sum_{n=0}^\infty \frac{(-1)^n}{(2n)!}2^{2n}x^{2n+1}\\
      &=& \sum_{n=0}^\infty \frac{(-1)^n2^{2n}}{(2n)!}x^{2n+1}
    \end{eqnarray*}
  \item Del item anterior tenemos que
  \begin{eqnarray*}
    f(x) &=& \sum_{n=0}^\infty \frac{(-1)^n2^{2n}}{(2n)!}x^{2n+1}\\
    &=& x-\frac{2^2}{2!}x^3+\frac{2^4}{4!}x^5-\frac{2^6}{6!}x^7+\cdots\\
    &=& x-2x^3+\frac{2}{3}x^5-\frac{4}{45}x^7+\cdots
  \end{eqnarray*}
  Luego, el polinomio de Taylor de grado 5 es
  $$T_5(x) = x-2x^3+\frac{2}{3}x^5$$
  Por lo tanto, la aproximación de $f(0.5)$ es
  \begin{eqnarray*}
    f(0.5) &\approx& T_5(0.5)\\
    &=& T_5\left(\frac{1}{2}\right)\\
    &=& \frac{1}{2}-2\left(\frac{1}{2}\right)^3+\frac{2}{3}\left(\frac{1}{2}\right)^5\\
    &=& \frac{1}{2}-2\left(\frac{1}{8}\right)+\frac{2}{3}\left(\frac{1}{32}\right)\\
    &=& \frac{1}{2}-\frac{1}{4}+\frac{1}{48}\\
    &=& \frac{24-12+1}{48}\\
    &=& \frac{13}{48}
  \end{eqnarray*}
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) } 
\begin{enumerate}[a)]
  \item Muestre $T:\Real^2\to\Real^2$ definida por
    $$T(x,y)=(4x-2y^2,2xy+1).$$
    no es una transformación lineal.
  \item Sea $T:\Real^2\to\Real^2$ una transformación lineal definida por
    $$T(x,y)=(3x+y,5x-7y)$$
    encuentre su matriz canónica.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Para demostrar que $T$ no es una transformación lineal, basta con encontrar un contraejemplo para alguna de las propiedades de las transformaciones lineales. Una opción es ver si $T(0,0)=0$. Si tomamos $(x,y)=(0,0)$, tenemos que 
  \begin{eqnarray*}
    T(0,0) &=& (4\cdot 0-2\cdot 0^2,2\cdot 0\cdot 0+1)\\
    &=& (0,1)
  \end{eqnarray*}
  Luego, como $T(0,0)\neq 0$, entonces $T$ no es una transformación lineal.

  \vspace{3mm}
  Otra respuesta puede ser comprobar si $T$ no cumple con la propiedad aditiva en algún par de vectores. Por ejemplo, si tomamos $(x,y)=(1,1)$ y $(x,y)=(1,-1)$, tenemos que
  \begin{eqnarray*}
    T(1,1) &=& (4\cdot 1-2\cdot 1^2,2\cdot 1\cdot 1+1) = (2,3)\\
    T(1,-1) &=& (4\cdot 1-2\cdot (-1)^2,2\cdot 1\cdot (-1)+1) = (2,-1).
  \end{eqnarray*}
  Luego, si $T$ fuera una transformación lineal, entonces debería cumplir que
  $$T(1,1)+T(1,-1) = T(1+1,1+(-1)) = T(2,0)$$
  Sin embargo, tenemos que
  $$T(1,1)+T(1,-1) = (2,3)+(2,-1) = (4,2)\neq (6,0) = T(2,0)$$
  Por lo tanto, $T$ no es una transformación lineal.

  \vspace{5mm}
  \item Para encontrar la matriz canónica de $T(x,y)=(3x+y,5x-7y)$, primero debemos encontrar la imagen de los vectores de la base canónica de $\Real^2$. Es decir, debemos encontrar $T(1,0)$ y $T(0,1)$.
  \begin{eqnarray*}
    T(1,0) &=& (3\cdot 1+0,5\cdot 1-7\cdot 0) = (3,5)\\
    T(0,1) &=& (3\cdot 0+1,5\cdot 0-7\cdot 1) = (1,-7)
  \end{eqnarray*}
  Luego, la matriz canónica de $T$ es
  $$[T] = \begin{bmatrix} 3 & 1\\ 5 & -7 \end{bmatrix}$$
\end{enumerate}
