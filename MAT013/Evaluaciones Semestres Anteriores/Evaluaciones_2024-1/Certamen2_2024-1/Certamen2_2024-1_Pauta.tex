%! Pregunta 1
\noindent\textbf{P1) } 
\begin{enumerate}[a)]
  \item Encuentre la serie de MacLaurin que representa a la función.
    $$f(x)=x\ln(1+x^2)$$
    Indicando el radio de convergencia.
  \item Encuentre la matriz de la transformación lineal $T:\Real^3\to\Real^2$ definida por
    $$T(x,y,z)=(x+2y+z,2x-y+3z)$$ 
\end{enumerate}


\vspace{0.25cm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Primero sabemos que
  $$\ln(1+x)=\sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{x^n}{n}$$
  luego
  \begin{eqnarray*}
    \ln(1+x^2) &=& \sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{(x^2)^n}{n}\\
    &=& \sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{x^{2n}}{n}
  \end{eqnarray*}
  ahora multiplicamos por $x$ y obtenemos
  \begin{eqnarray*}
    x\ln(1+x^2) &=& x\sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{x^{2n}}{n}\\
    &=& \sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{x^{2n+1}}{n}
  \end{eqnarray*}
  por lo tanto, la serie de MacLaurin de $f(x)$ es
  $$f(x)=\sum_{n=1}^{\infty}(-1)^{n-1}\dfrac{x^{2n+1}}{n}$$
  \item Para encontrar la matriz canónica de la transformación lineal, calculamos la imagen de los vectores canónicos
  \begin{eqnarray*}
    T(1,0,0) &=& (1,2)\\
    T(0,1,0) &=& (2,-1)\\
    T(0,0,1) &=& (1,3)
  \end{eqnarray*}
  por lo tanto, la matriz de la transformación lineal es
  $$\begin{pmatrix}
    1 & 2 & 1\\
    2 & -1 & 3
  \end{pmatrix}$$
\end{enumerate}

  $$F(x,y)=3-x^2-y^2$$

  $$ x^2+4y^2-z^2-2x-16y+2z = 0$$

\noindent\textbf{Respuesta:}

  $$3-x^2-y^2=k$$
  reordenando obtenemos
  $$x^2+y^2=3-k$$

  $$z=3-y^2$$
  cuyo gráfico es
  \begin{center}
    \includegraphics[scale=0.7]{Certamen2_Corte_Transversal.png}
  \end{center}
  
  Finalmente, el gráfico de la superficie es
  \begin{center}
    \includegraphics[scale=1]{Certamen2_Superficie_parametrica.png}
  \end{center}
  \newpage
  \item Para identificar la superficie, completamos cuadrados obteniendo
  \begin{eqnarray*}
    x^2+4y^2-z^2-2x-16y+2z &=& 0\\
    x^2-2x+4y^2-16y-z^2+2z &=& 0\\
    (x-1)^2-1+4(y-2)^2-16-1-(z-1)^2+1 &=& 0\\

\end{enumerate}


%! Pregunta 3
\newpage
\noindent\textbf{P3) } Analice la continuidad de la función
$$G(x,y)=\begin{cases}
  \dfrac{3x^2y}{x^2+y^2} & \text{si } (x,y)\neq (0,0)\\
  0 & \text{si } (x,y)=(0,0)
\end{cases}$$


\vspace{0.25cm}
\noindent\textbf{Respuesta:} Primero notamos que la función es continua en todos los puntos $(x,y)\neq (0,0)$, ya que es una fracción de polinomios sin indeterminaciones. Para analizar la continuidad en $(0,0)$, consideramos el límite
$$\lim_{(x,y)\to (0,0)}G(x,y)$$
acotando el valor absoluto de la función obtenemos
\begin{eqnarray*}
  \abs{G(x,y)} &=& \abs{\dfrac{3x^2y}{x^2+y^2}}\\
  &=& \dfrac{3x^2\abs{y}}{x^2+y^2}\\
  &\leq& \dfrac{3x^2\abs{y}}{x^2}\\
  &=& 3\abs{y}
\end{eqnarray*}
luego, como
$$\lim_{(x,y)\to (0,0)}3\abs{y}=0$$
entonces
$$\lim_{(x,y)\to (0,0)}G(x,y)=0=G(0,0)$$
por lo tanto, la función es continua en todo $\Real^2$.
