%! Pregunta 1
\noindent\textbf{P1) } \textsf{[ 30 Pts. ]}
\begin{enumerate}[a)]
  \item Usando derivación implícita, encuentre $\dfrac{\partial z}{\partial x}$ y $\dfrac{\partial z}{\partial y}$ si $z$ está definida implícitamente por la ecuación
  $$x^2+2y^2+3z^2=6$$
  \item Encuentre la ecuación del plano tangente a la superficie
  $$x^2+2y^2+3z^2=6$$
  en el punto $(1,1,1)$.
\end{enumerate}

\vspace{2mm}
\noindent\textbf{Respuesta:} 
\begin{enumerate}[a)]
  \item Si escribimos la ecuación como $F(x,y,z)=0$, en este caso
  $$F(x,y,z)=x^2+2y^2+3z^2-6=0$$
  entonces, por derivación implícita, tenemos que
  $$\dfrac{\partial z}{\partial x}=-\dfrac{F_x}{F_z}$$
  y
  $$\dfrac{\partial z}{\partial y}=-\dfrac{F_y}{F_z}$$
  En este caso, obtenemos
  $$F_x=2x\qquad F_y=4y\qquad F_z=6z$$
  Luego,
  $$\dfrac{\partial z}{\partial x}=-\dfrac{2x}{6z}=-\dfrac{x}{3z}$$
  y
  $$\dfrac{\partial z}{\partial y}=-\dfrac{4y}{6z}=-\dfrac{2y}{3z}$$
  \item Para encontrar la ecuación del plano tangente, primero encontramos las derivadas parciales de $F$ en el punto $(1,1,1)$, es decir,
  $$F_x(1,1,1)=2\qquad F_y(1,1,1)=4\qquad F_z(1,1,1)=6$$
  Luego, la ecuación del plano tangente es
  $$2(x-1)+4(y-1)+6(z-1)=0$$
  que simplificando queda
  $$2x+4y+6z=12$$
\end{enumerate}


%! Pregunta 2
\newpage
\noindent\textbf{P2) } \textsf{[ 40 Pts. ]}
\begin{enumerate}[a)]
  \item Encuentre y clasifique los puntos críticos de la función
  % $$f(x,y)=3x-x^3-2y^2+y^4$$
  $$f(x,y)=y^3+3x^2y-6x^2-6y^2+2$$
  \item Encuentre los valores máximos y mínimos de la función
  $$f(x,y)=3x+2y$$
  sujeta a la restricción
  $$x^2+y^2=1$$
\end{enumerate}

\vspace{2mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Para encontrar los puntos críticos, primero encontramos las derivadas parciales de $f$ y las igualamos a cero, es decir,
  $$f_x=6xy-12x$$
  $$f_y=3y^2+3x^2-12y$$
  Luego, igualamos a cero y resolvemos el sistema
  $$\begin{array}{rcl|}
    6xy-12x&=&0\\
    3y^2+3x^2-12y&=&0\\
    \hline
  \end{array}$$
  cuyas soluciones son $(0,0)$, $(0,4)$, $(2,2)$ y $(-2,2)$. Ahora calculamos las segundas derivadas y evaluamos en los puntos críticos para clasificarlos. $$f_{xx}=6y-12\qquad f_{yy}=6y-12\qquad f_{xy}=f_{yx}=6x$$
  Evaluando en los puntos críticos, tenemos que
  \begin{enumerate}[i.]
    \item En $(0,0)$, 
    $$H=f_{xx}(0,0)f_{yy}(0,0)-(f_{xy}(0,0))^2=(-12)(-12)-(0)^2>0$$
    y $f_{xx}(0,0)<0$, por lo que es un máximo relativo.
    \item En $(0,4)$,
    $$H=f_{xx}(0,4)f_{yy}(0,4)-(f_{xy}(0,4))^2=(12)(12)-(0)^2>0$$
    y $f_{xx}(0,4)>0$, por lo que es un mínimo relativo.
    \item En $(2,2)$,
    $$H=f_{xx}(2,2)f_{yy}(2,2)-(f_{xy}(2,2))^2=(0)(0)-(12)^2<0$$
    por lo que es un punto silla.
    \item En $(-2,2)$,
    $$H=f_{xx}(-2,2)f_{yy}(-2,2)-(f_{xy}(-2,2))^2=(0)(0)-(-12)^2<0$$
    por lo que es un punto silla.
  \end{enumerate}
  \newpage
  \item Para encontrar los valores máximos y mínimos, usamos el método de los multiplicadores de Lagrange. Definimos la función
  $$L(x,y,\lambda)=3x+2y-\lambda(x^2+y^2-1)$$
  y encontramos las derivadas parciales
  $$L_x=3-2\lambda x=0$$
  $$L_y=2-2\lambda y=0$$
  $$L_\lambda=x^2+y^2-1=0$$
  Resolviendo el sistema, obtenemos los puntos críticos
  $$\left(\dfrac{3}{\sqrt{13}},\dfrac{2}{\sqrt{13}}\right)\text{ y }\left(-\dfrac{3}{\sqrt{13}},-\dfrac{2}{\sqrt{13}}\right)$$
  Evaluando la función en estos puntos, obtenemos
  $$f\left(\dfrac{3}{\sqrt{13}},\dfrac{2}{\sqrt{13}}\right)=\frac{13}{\sqrt{13}}=\sqrt{13}$$
  $$f\left(-\dfrac{3}{\sqrt{13}},-\dfrac{2}{\sqrt{13}}\right)=-\frac{13}{\sqrt{13}}=-\sqrt{13}$$
  por lo que el máximo se alcanza en $\left(\dfrac{3}{\sqrt{13}},\dfrac{2}{\sqrt{13}}\right)$ y el mínimo en $\left(-\dfrac{3}{\sqrt{13}},-\dfrac{2}{\sqrt{13}}\right)$.
\end{enumerate}

\noindent\textbf{P3) } \textsf{[ 30 Pts. ]}
\begin{enumerate}[a)]
  \item Calcule la integral
  $$\iint_D\ x^2+y^2\ dA$$
  Donde $D$ es la región del primer cuadrante limitada por las rectas $y=x$, $y=0$ y por la circunferencia $x^2+y^2=4$.
  \item Calcule la integral
  $$\iiint_E\ y\ dV$$
  Donde $E=\set{(x,y,z)\in\Real^3:\ 0\leq x\leq 3\ \wedge\ 0\leq y\leq x\ \wedge\ x-y\leq z\leq x+y}$
\end{enumerate}


\vspace{2mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Graficando la región $D$,
  \vspace{-1cm}

  % \begin{figure}[h]
    \begin{center}
      \includegraphics[scale=0.6]{Certamen3-3a.png}
    \end{center}
  % \end{figure}
  \vspace{-1cm}

  Usando coordenadas polares, el dominio $D$ se puede describir como
  $$D^{\ast}=\set{(r,\theta)\in\Real^2:\ 0\leq r\leq 2\ \wedge\ 0\leq \theta\leq \frac{\pi}{4}}$$
  Luego, la integral se puede escribir como
  \begin{eqnarray*}
    \iint_D x^2+y^2 dA&=&\int_0^{\frac{\pi}{4}}\int_0^2 r^2\cdot r\ drd\theta\\
    &=&\int_0^{\frac{\pi}{4}}\int_0^2 r^3\ drd\theta\\
    &=&\int_0^{\frac{\pi}{4}}\left[\dfrac{r^4}{4}\right]_0^2\ d\theta\\
    &=&\int_0^{\frac{\pi}{4}}4\ d\theta\\
    % &=&4\left[\theta\right]_0^{\frac{\pi}{4}}\\
    &=&\pi
  \end{eqnarray*}
  \item Dada la descripción del dominio $E$, la integral se puede escribir como
  \begin{eqnarray*}
    \iiint_E y dV&=&\int_0^3\int_0^x\int_{x-y}^{x+y} y\ dzdydx\\
    % &=&\int_0^3\int_0^x\left[yz\right]_{x-y}^{x+y} dydx\\
    &=&\int_0^3\int_0^x\left[(x+y)-(x-y)\right]y dydx\\
    &=&\int_0^3\int_0^x 2y^2 dydx\\
    &=&\int_0^3\left[\dfrac{2y^3}{3}\right]_0^x dx\\
    &=&\int_0^3 \dfrac{2x^3}{3} dx\\
    &=&\left[\dfrac{x^4}{6}\right]_0^3\\
    % &=&\dfrac{3^4}{6}\\
    % &=&\dfrac{81}{6}\\
    &=&\dfrac{27}{2}
  \end{eqnarray*}
\end{enumerate}
