%! Pregunta 1
\noindent\textbf{P1) } Determine si las siguientes sucesiones convergen o divergen, y en caso de converger, encuentre su límite.
\begin{enumerate}[a)]
  \item $a_n = \dfrac{3n^2-1}{n+1}$
  \item $b_n = \sen\left(\dfrac{\pi n^2+3n}{2n^2 + 1}\right)$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item 
  \begin{eqnarray*}
    \lim_{n\to\infty} \dfrac{3n^2-1}{n+1} &=& \lim_{n\to\infty} \dfrac{3n^2-1}{n+1}\cdot\dfrac{\frac{1}{n}}{\frac{1}{n}}\\ 
    &=& \lim_{n\to\infty} \dfrac{3n-\frac{1}{n}}{1+\frac{1}{n}}\\
    &=& \infty
  \end{eqnarray*}
  \item
  \begin{eqnarray*}
    \lim_{n\to\infty} \sen\left(\dfrac{\pi n^2+3n}{2n^2 + 1}\right) 
      &=& \sen\left(\lim_{n\to\infty}\dfrac{\pi n^2+3n}{2n^2 + 1}\right)\\
      &=& \sen\left(\lim_{n\to\infty}\dfrac{\pi n^2+3n}{2n^2 + 1}\cdot\dfrac{\frac{1}{n^2}}{\frac{1}{n^2}}\right)\\
      &=& \sen\left(\lim_{n\to\infty}\dfrac{\pi +\frac{3}{n}}{2 + \frac{1}{n^2}}\right)\\
      &=& \sen\left(\dfrac{\pi}{2}\right)\\
      &=& 1
  \end{eqnarray*}
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) } Determine, indicando el criterio utilizado, si las siguientes series convergen o divergen.
\begin{enumerate}[a)]
  \item $\displaystyle\sum_{n=2}^{\infty} \dfrac{1}{n(\ln(n))^2}$
  \item $\displaystyle\sum_{n=1}^{\infty} \dfrac{\sqrt{n^2+2}}{n^4+n^2}$
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Usando el criterio de la integral, debemos analizar la convergencia de la integral
  \begin{eqnarray*}
    \int_{1}^{\infty} \dfrac{1}{x(\ln(x))^2}dx 
      &=& \lim_{b\to\infty}\int_{2}^{b} \dfrac{1}{x(\ln(x))^2}dx\\
      &=& \lim_{b\to\infty}\int_{\ln(2)}^{\ln(b)} \dfrac{1}{u^2}du\\
      &=& \lim_{b\to\infty}\left[-\dfrac{1}{u}\right]_{\ln(2)}^{\ln(b)}\\
      &=& \lim_{b\to\infty}\left[-\dfrac{1}{\ln(b)} + \dfrac{1}{\ln(2)}\right]\\
      &=& \dfrac{1}{\ln(2)}
  \end{eqnarray*}
  Por lo tanto, como la integral converge, la serie también converge.\\

  \item Analizando intuitivamente el término general de la serie vemos que se puede comparar con la serie $\displaystyle\sum_{n=1}^{\infty} \dfrac{1}{n^3}$, usando el criterio de la comparación en el límite, se tiene que
  \begin{eqnarray*}
    \lim_{n\to\infty} \dfrac{\frac{\sqrt{n^2+2}}{n^4+n^2}}{\frac{1}{n^3}} 
      &=& \lim_{n\to\infty} \dfrac{n^3\sqrt{n^2+2}}{n^4+n^2}\\
      &=& \lim_{n\to\infty} \dfrac{n\sqrt{n^2+2}}{n^2+1}\\
      &=& \lim_{n\to\infty} \dfrac{n\sqrt{n^2+2}}{n^2+1}\cdot\dfrac{\frac{1}{n^2}}{\frac{1}{n^2}}\\
      &=& \lim_{n\to\infty} \dfrac{\left(\frac{\sqrt{n^2+2}}{n}\right)}{1+\frac{1}{n^2}}\\
      &=& \lim_{n\to\infty} \dfrac{\sqrt{1+\frac{2}{n^2}}}{1+\frac{1}{n^2}}\\
      &=& 1
  \end{eqnarray*}
  Por lo tanto, como la serie $\displaystyle\sum_{n=1}^{\infty} \dfrac{1}{n^3}$ converge, la serie $\displaystyle\sum_{n=1}^{\infty} \dfrac{\sqrt{n^2+2}}{n^4+n^2}$ también converge.
\end{enumerate}
