%! Pregunta 1
\noindent\textbf{P1) } \textsf{[50 pts.]} Determine si las siguientes funciones son transformaciones lineales.
\begin{enumerate}[a)]
  \item $T:\Real^2\to\Real^2$ tal que $T(x,y)=(2x+3y,x-2y)$.
  \item $T:\Real^3\to\Real^3$ tal que $T(x,y,z)=(x+y+z,xy,xz)$.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Para que $T$ sea una transformación lineal, debe cumplir las siguientes propiedades.
    \begin{enumerate}[i)]
      \item $T(u+v)=T(u)+T(v)$.\\[2mm]
        Sean $u=(x_1,y_1)$ y $v=(x_2,y_2)$.
        \begin{eqnarray*}
          T(u+v)
            &=& T((x_1,y_1)+(x_2,y_2)) \\
            &=& T((x_1+x_2),(y_1+y_2)) \\
            &=& (2(x_1+x_2)+3(y_1+y_2),(x_1+x_2)-2(y_1+y_2)) \\
            &=& ((2x_1+3y_1)+(2x_2+3y_2),(x_1-2y_1)+(x_2-2y_2)) \\
            &=& (2x_1+3y_1,x_1-2y_1)+(2x_2+3y_2,x_2-2y_2) \\
            &=& T(u)+T(v).
        \end{eqnarray*}
      \item $T(\alpha\vec{u})=\alpha T(\vec{u})$.\\[2mm]
        Sean $u=(x,y)$ y $\alpha\in\Real$.
        \begin{eqnarray*}
          T(\alpha u)
            &=& T(\alpha(x,y)) \\
            &=& T((\alpha x,\alpha y)) \\
            &=& (2(\alpha x)+3(\alpha y),(\alpha x)-2(\alpha y)) \\
            &=& (\alpha(2x+3y),\alpha(x-2y)) \\
            &=& \alpha(2x+3y,x-2y) \\
            &=& \alpha T(u).
        \end{eqnarray*}
    \end{enumerate}
    Por lo tanto, $T$ es una transformación lineal.
  \newpage
  \item Mostraremos que $T$ no es una transformación lineal, para esto basta que una propiedad no se cumpla con vectores concretos.\\[2mm]
    Sean $u=(1,0,0)$ y $v=(0,1,0)$.
    \begin{eqnarray*}
      T(u+v)
        &=& T((1,0,0)+(0,1,0)) \\
        &=& T((1,1,0)) \\
        &=& (1+1+0,1\cdot 1,1\cdot 0) \\
        &=& (2,1,0).
    \end{eqnarray*}
    Por otro lado,
    \begin{eqnarray*}
      T(u)+T(v)
        &=& T((1,0,0))+T((0,1,0)) \\
        &=& (1+0+0,1\cdot 0,1\cdot 0)+(0+1+0,0\cdot 1,0\cdot 0) \\
        &=& (1,0,0)+(1,0,0) 
    \end{eqnarray*}
    Como $T(u+v)\neq T(u)+T(v)$ entonces $T$ no es una transformación lineal.
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) } \textsf{[50 pts.]} Determine si los siguientes conjuntos son abiertos, cerrados, compactos o conexos en $\Real^2$.
\begin{enumerate}[a)]
  \item $A=\set{(x,y)\in\Real^2:\ 4x^2+25y^2 \leq 100 \wedge x+y \leq 1}$.
  \item $B=\set{(x,y)\in\Real^2:\ 4x^2+y^2<4 \wedge \abs{y} > 1}$.
\end{enumerate}

\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item Gráficamente la region se ve de la siguiente manera.
  \vspace{-10mm}
  \begin{center}
    \includegraphics[scale=0.4]{./imagenes/Pregunta2a.png}
  \end{center}
    \begin{itemize}
      \item $A$ no es abierto, pues el punto $(-5,0)$ pertenece a $A$ y no es un punto interior de $A$.
      \item $A$ es cerrado pues contiene a todos sus puntos frontera.
      \item $A$ es compacto, porque es cerrado y acotado por la circunferencia de radio $5$ centrada en el origen.
      \item $A$ es conexo, ya que no se puede separar ni hay puntos aislados.
    \end{itemize}
  \item Gráficamente la region se ve de la siguiente manera.
  \vspace{-10mm}
  \begin{center}
    \includegraphics[scale=0.5]{./imagenes/Pregunta2b.png}
  \end{center}
    \begin{itemize}
      \item $B$ es abierto, porque todos sus puntos son interiores.
      \item $B$ no es cerrado, el punto $(0,2)$ pertenece a la frontera de $B$ pero no está en $B$.
      \item $B$ no es compacto, pues no es cerrado.
      \item $B$ no es conexo, pues se puede separar usando los abiertos dados por las inecuaciones $y>\frac{1}{2}$ e $y<-\frac{1}{2}$.
    \end{itemize}
\end{enumerate}
