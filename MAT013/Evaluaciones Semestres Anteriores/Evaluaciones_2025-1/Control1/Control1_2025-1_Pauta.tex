%! Pregunta 1
\noindent\textbf{P1) } Determine si las siguientes sucesiones convergen o divergen, y en caso de converger, encuentre su límite.
\begin{enumerate}[a)]
  \item $a_n = \tan\left(\dfrac{\pi n^2+n}{4n^2 - 1}\right)$
  \item $b_n = \dfrac{n-2}{\sqrt{9n^2+4}}$
\end{enumerate}


\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item 
    \begin{eqnarray*}
      \lim_{n\to\infty} \tan\left(\dfrac{\pi n^2+n}{4n^2 - 1}\right) 
      &=& \tan\left(\lim_{n\to\infty}\dfrac{\pi n^2+n}{4n^2 - 1}\right) \\
      &=& \tan\left(\lim_{n\to\infty}\dfrac{\pi n^2+n}{4n^2 - 1}\cdot\dfrac{\frac{1}{n^2}}{\frac{1}{n^2}}\right) \\
      &=& \tan\left(\lim_{n\to\infty}\dfrac{\pi + \dfrac{1}{n}}{4 - \dfrac{1}{n^2}}\right) \\
        &=& \tan\left(\dfrac{\pi}{4}\right) \\
        &=& 1
    \end{eqnarray*}
  \item 
    \begin{eqnarray*}
      \lim_{n\to\infty} \dfrac{n-2}{\sqrt{9n^2+4}} 
      &=& \lim_{n\to\infty} \dfrac{n-2}{\sqrt{9n^2+4}}\cdot\dfrac{\frac{1}{n}}{\frac{1}{n}} \\
      &=& \lim_{n\to\infty} \dfrac{1-\frac{2}{n}}{\left(\frac{\sqrt{9n^2+4}}{n}\right)}\\
      &=& \lim_{n\to\infty} \dfrac{1-\frac{2}{n}}{\sqrt{9+\frac{4}{n^2}}} \\
      &=& \dfrac{1-0}{\sqrt{9+0}} \\
      &=& \dfrac{1}{3}
    \end{eqnarray*}
\end{enumerate}

%! Pregunta 2
\noindent\textbf{P2) } Determine, indicando el criterio utilizado, si las siguientes series convergen o divergen.
\begin{enumerate}[a)]
  \item $\displaystyle\sum_{n=2}^{\infty} \ln\left(\dfrac{2n^2-n+2}{n^2+2n+2}\right)$
  \item $\displaystyle\sum_{n=1}^{\infty} \dfrac{\sqrt{n^2+1}}{n^3+2n^2}$
\end{enumerate}


\vspace{5mm}
\noindent\textbf{Respuesta:}
\begin{enumerate}[a)]
  \item si calculamos
    \begin{eqnarray*}
      \lim_{n\to\infty} \ln\left(\dfrac{2n^2-n+2}{n^2+2n+2}\right)
        &=& \ln\left(\lim_{n\to\infty}\dfrac{2n^2-n+2}{n^2+2n+2}\right)\\
        &=& \ln\left(\lim_{n\to\infty}\dfrac{2-\frac{1}{n}+\frac{2}{n^2}}{1+\frac{2}{n}+\frac{2}{n^2}}\right)\\
        &=& \ln\left(\dfrac{2-0+0}{1+0+0}\right)\\
        &=& \ln(2)
    \end{eqnarray*}
    como $\lim\limits_{n\to\infty} \ln\left(\dfrac{2n^2-n+2}{n^2+2n+2}\right) = \ln(2) \neq 0$, entonces la serie $\displaystyle\sum_{n=2}^{\infty} \ln\left(\dfrac{2n^2-n+2}{n^2+2n+2}\right)$ diverge.
  \item Analizando el termino general de la serie podemos ver que nos conviene usar comparación en el límite con $\sum_{n=1}^{\infty} \dfrac{1}{n^2}$, para esto calculamos.
    \begin{eqnarray*}
      \lim_{n\to\infty} \dfrac{\frac{\sqrt{n^2+1}}{n^3+2n^2}}{\frac{1}{n^2}}
        &=& \lim_{n\to\infty} \dfrac{n^2\sqrt{n^2+1}}{n^3+2n^2}\\
        &=& \lim_{n\to\infty} \dfrac{\sqrt{n^2+1}}{n+2}\\
        &=& \lim_{n\to\infty} \dfrac{\sqrt{n^2+1}}{n+2}\cdot\dfrac{\frac{1}{n}}{\frac{1}{n}}\\
        &=& \lim_{n\to\infty} \dfrac{\sqrt{1+\frac{1}{n^2}}}{1+\frac{2}{n}}\\
        &=& \dfrac{\sqrt{1+0}}{1+0}\\
        &=& 1
    \end{eqnarray*}
    Por lo tanto, como $\sum_{n=1}^{\infty} \dfrac{1}{n^2}$ es convergente, por el criterio de comparación, la serie $\sum_{n=1}^{\infty} \dfrac{\sqrt{n^2+1}}{n^3+2n^2}$ también es convergente.
\end{enumerate}
