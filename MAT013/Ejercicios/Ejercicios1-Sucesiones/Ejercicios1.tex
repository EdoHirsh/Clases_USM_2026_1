%! Ejercicios
\begin{enumerate}
  \item Calcule los siguientes limites de sucesiones
  \begin{enumerate}
    \item $\displaystyle\lim_{n\to\infty} \frac{3n^2+1}{4n^2+2n+1}$
    \item $\displaystyle\lim_{n\to\infty} \frac{2n^2+5n+1}{2n+1}$
    \item $\displaystyle\lim_{n\to\infty} \frac{(n+1)^{n+1}}{n^n}$
    \item $\displaystyle\lim_{n\to\infty} \sqrt[n]{\frac{n^n(n+1)!}{(n+1)^{n+1}n!}}$
  \end{enumerate}
  \vspace{2mm}
  \item Determine si las siguientes series convergen absolutamente, convergen condicionalmente o divergen.
  \begin{enumerate}
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{(-5)^n}{n!}$
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{2^n}{n^2}$
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{(-3)^n}{(2n+1)!}$
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{\cos(2n)}{n^4}$
  \end{enumerate}
  \vspace{2mm}
  \item Encuentre el radio y el intervalo de convergencia de las siguientes series.
  \begin{enumerate}
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{(-3)^n}{n\sqrt{n}}x^n$
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{(x-3)^n}{n^n}$
    \item $\displaystyle\sum_{n=1}^{\infty} \frac{(5x-4)^n}{n^3}$
    \item $\displaystyle\sum_{n=0}^{\infty} \frac{(-1)^n}{(2n+1)!}x^n$
  \end{enumerate}
  \vspace{2mm}
  \newpage
  \item Encuentre una representación en serie de potencias de las siguientes funciones.
  \begin{enumerate}
    \item $\dfrac{x}{9+x^2}$
    \item $\ln(1+x)$
    \item $\dfrac{1+x}{1-x}$
    \item $\dfrac{x}{2x^2+1}$
  \end{enumerate}
  \vspace{2mm}
  \item Determine una Serie de Taylor y su radio de convergencia para las siguientes funciones.
    \begin{enumerate}
      \item $x\sen(x^2)$
      \item $x\ln(1+x^2)$
      \item $x^3e^{x^2}$
      \item $\dfrac{x}{1+x^2}$
    \end{enumerate}
\end{enumerate}
