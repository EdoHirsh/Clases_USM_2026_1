\begin{frame}%{Parte 1}
  \begin{enumerate}
    \item Calcule.
      \begin{enumerate}
        \item $\lim\limits_{n\to\infty}\ln\left(\dfrac{2n^2+1}{n^3+n^2}\right)$
        \item $\displaystyle\sum_{n=1}^{\infty}\dfrac{(-10)^{n+1}}{15^{n-1}}$
      \end{enumerate}
    \vspace{2mm}
    \item Indique usando el criterio adecuado si las siguientes series convergen o divergen.
      \begin{enumerate}
        \item $\displaystyle\sum_{n=0}^{\infty}\dfrac{(-3)^n}{n!}$
        \item $\displaystyle\sum_{n=1}^{\infty}\dfrac{2n+1}{\sqrt{n^4+2n}}$
      \end{enumerate}
    \vspace{2mm}
    \item 
      \begin{enumerate}
        \item Encuentre el intervalo de convergencia de la serie 
          $$\displaystyle\sum_{n=0}^{\infty}\dfrac{(3x)^n}{n^2+1}$$
        \item Encuentre la serie de Maclaurin de la función $f(x)=x\cos(2x)$.
        \item Use un polinomio de Taylor de grado 5 para aproximar el valor de $f(0.5)$.
      \end{enumerate}
  \end{enumerate}
\end{frame}
