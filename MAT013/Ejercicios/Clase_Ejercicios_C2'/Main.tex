    \end{cases}$$
    Calcule.
    \begin{multicols}{3}
      \begin{enumerate}
        \item $\displaystyle\lim_{(x,y)\to(1,1)}f(x,y)$
        \item $\displaystyle\lim_{(x,y)\to(1,-1)}f(x,y)$

\begin{frame}%{Parte 2}
  \begin{enumerate}
    \setcounter{enumi}{2}
    \item Dada la función
        $$f(x,y,z) = x^3-xy+y^2+xz+z^3$$
      \begin{enumerate}
        \item Calcule las derivadas parciales de $f$.
        \item Encuentre la dirección de máximo crecimiento de $f$ en el punto $(1,0,2)$.
        \item Encuentre la ecuación del plano tangente a la superficie de ecuación
          $$f(x,y,z)=1$$
          en el punto $(1,1,0)$.
      \end{enumerate}
  \end{enumerate}
\end{frame}
