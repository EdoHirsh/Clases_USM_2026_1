%! Ejercicios
\begin{enumerate}
  \item Decida si las siguientes funciones son transformaciones lineales. En caso afirmativo, determine su matriz asociada, en caso contrario, justifique.
    \begin{enumerate}
      \item $T:\Real^2\to\Real^2$ dada por $T(x,y) = (x+2y, x-3y)$
      \item $T:\Real^2\to\Real^2$ dada por $T(x,y) = (x^2-y^2, y^2+x)$
      \item $T:\Real^2\to\Real^3$ dada por $T(x,y) = (x+y, x-y, 2x-1)$
      \item $T:\Real^3\to\Real^2$ dada por $T(x,y,z) = (x+y-z, \sqrt{x^2-y^2})$
      \item $T:\Real^3\to\Real^3$ dada por $T(x,y,z) = (2x-3y+z, x+2y-3z, 3x-2y+z)$
      \item $T:\Real^3\to\Real^3$ dada por $T(x,y,z) = (x^2+2xy+y^2, x^2-y^2, z)$
    \end{enumerate}
  \vspace{2mm}
  \item En los siguientes conjuntos, grafique el conjunto y determine si son abiertos, cerrados, compactos, conexos, disconexos, regiones o ninguno de los anteriores.
    \begin{enumerate}
      \item $\set{(x,y)\in\Real^2\ :\ x^2+y^2<1\wedge y<\abs{x}}$
      \item $\set{(x,y)\in\Real^2\ :\ x-2y<1 \wedge x+y\geq 1}$
      \item $\set{(x,y)\in\Real^2\ :\ 4x^2+9y^2\leq 36 \wedge \abs{x+y}\geq 1}$
      \item $\set{(x,y)\in\Real^2\ :\ xy\geq 1 \wedge x^2+y^2\leq 4 \wedge x>0}$
    \end{enumerate}
  \vspace{2mm}
  \item Use curvas de nivel y cortes transversales, para hacer un bosquejo del gráfico de las siguientes funciones.
    \begin{enumerate}
      \item $f(x,y)=1-x^2-4y^2$.
      \item $f(x,y)=1-x^2$.
      \item $f(x,y)=x^2-y^2$.
      \item $f(x,y)=\sqrt{36-4x^2-9y^2}$.
    \end{enumerate}
  \vspace{2mm}
  \item Escriba las siguientes ecuaciones en forma estándar y realice un bosquejo de la superficie cuádrica correspondiente.
    \begin{enumerate}
      \item $x^2+2z^2-2x-8z-7=0$
      \item $4x^2-y^2+z^2+8x+4y+2z+3=0$
      \item $x^2-y^2-2z^2+6x+4y+4z+1=0$
      \item $x^2+z^2-2x-4y-2z-4=0$
      \item $x^2-z^2-2x-y-2z-1=0$
      \item $x^2+4y^2+25z^2-2x-16y-100z+17=0$
    \end{enumerate}
  \vspace{2mm}
  \newpage
  \item Calcule el límite de las siguientes funciones o muestre que no existe.
    \begin{enumerate}
      \item $\lim\limits_{(x,y)\to(1,2)}\dfrac{x^2y^2-4xy}{x^2+y^2-3x-6y+2}$
      \item $\lim\limits_{(x,y)\to(0,0)}\cos\left(\dfrac{x^2+\pi}{x^2+y^2-2}\right)$
      \item $\lim\limits_{(x,y)\to(0,0)}\dfrac{x^2y^2}{\sqrt{x^2+y^2+1}-1}$
      \item $\lim\limits_{(x,y)\to(0,0)}\dfrac{x^4-y^4}{x^2+y^2}$
      \item $\lim\limits_{(x,y)\to(0,0)}\dfrac{x^2\sen^2(y)}{x^2+2y^2}$
      \item $\lim\limits_{(x,y)\to(0,0)}\dfrac{x^2y^4}{x^2+y^8}$
    \end{enumerate}
  \vspace{2mm}
  \item Analice la continuidad de 
    $$f(x,y)=\begin{cases}
      \dfrac{x^4y^2}{x^4+y^4} & \text{si } (x,y)\neq(0,0)\\
      0 & \text{si } (x,y)=(0,0)
    \end{cases}$$
\end{enumerate}
